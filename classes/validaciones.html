import { FormControl } from "@angular/forms";



export class validaciones {


    constructor() {   

    }

    esBlanco(event: KeyboardEvent) {
        return false;
    }
    

    esNumerico(event: KeyboardEvent) {
        var charCode = (event.which) ? event.which : event.keyCode;
        // Only Numbers 0-9
        if ((charCode < 48 || charCode > 57)) {
        event.preventDefault();
        return false;
        } else {
        return true;
        }
    }
    
    esNumericoPositivo(event: KeyboardEvent) {
        var regex = new RegExp("[1-9]");
        var code = !event.charCode ? event.which : event.charCode;
        var key = String.fromCharCode(code);
    
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    esNumericoMayorIgualCero(event: KeyboardEvent) {
        var regex = new RegExp("[0-9]");
        var code = !event.charCode ? event.which : event.charCode;
        var key = String.fromCharCode(code);
    
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    esAlfanumericoConEspeciales(event: KeyboardEvent, min: number) {
        let regex;
        if (min)
            regex = new RegExp("^[0-9a-záéíóúüñ #$%()*+-./:;=?@_°¿]+$");
    
        if(!min)
            regex = new RegExp("^[0-9a-zA-ZáéíóúÁÉÍÓÚüÜñÑ #$%()*+-./:;=?@_°¿]+$");
    
        let code = !event.charCode ? event.which : event.charCode;
        let key = String.fromCharCode(code);
    
        if (code == 13)
            return true;
    
        if (!regex?.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    esTelefono(event: KeyboardEvent, control: FormControl) {
        var valor = control.value();
        var regex = new RegExp("^[1-9]{1}[0-9]{0,9}$");
        var code = !event.charCode ? event.which : event.charCode;
        var key = String.fromCharCode(code);
    
        if (!regex.test(valor + key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    esAlfanumericoGuion(event: KeyboardEvent) {
        var regex = new RegExp("^[0-9a-zA-ZáéíóúÁÉÍÓÚüÜñÑ -]+$");
        var code = !event.charCode ? event.which : event.charCode;
        var key = String.fromCharCode(code);
    
        if (code == 13)
            return true;
    
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    esAlfanumerico(event: KeyboardEvent) {
        var regex = new RegExp("^[0-9a-zA-Z áéíóúÁÉÍÓÚüÜñÑ]+$");
        var key = String.fromCharCode(!event.charCode ? event.which : event.charCode);
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    sonVaIidosCorreo(event: KeyboardEvent){
        var regex = new RegExp("^[0-9a-zA-Z @_.-]+$");
        var key = String.fromCharCode(!event.charCode ? event.which : event.charCode);
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    soloLetras(event: KeyboardEvent) {
        var regex = new RegExp("^[a-zA-Z áéíóúÁÉÍÓÚüÜñÑ]+$");
        var key = String.fromCharCode(!event.charCode ? event.which : event.charCode);
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
        return undefined;
    }

    esMayorA(event: KeyboardEvent, value: number, limit: number) {
        if(!this.esNumerico(event)){
            return false;
        }
        return value < limit
    }

    esMayorOIgualA(event: KeyboardEvent, value: number, limit: number) {
        if(!this.esNumerico(event)){
            return false;
        }
        return value <= limit
    }

}
