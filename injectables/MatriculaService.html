import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class MatriculaService {

  private matricula = new BehaviorSubject<string>('');
  matricula$ = this.matricula.asObservable();

  // TODO: tipar matriculaDetails
  private matriculaDetails = new BehaviorSubject<any | null>(null);
  matriculaDetails$ = this.matriculaDetails.asObservable();

  private nuevoReporteValido = new BehaviorSubject<boolean>(false);
  nuevoReporteValido$ = this.nuevoReporteValido.asObservable();

  private authMode = new BehaviorSubject<boolean>(false);
  authMode$ = this.authMode.asObservable();

  constructor() { }

  updateMatricula(value: string) {
    this.matricula.next(value);
  }

  updateMatriculaDetails(matriculaDetails: any | null) {
    this.matriculaDetails.next(matriculaDetails);
  }

  updateNuevoReporteValido(value: boolean) {
    this.nuevoReporteValido.next(value);
  }

  updateAuthMode(authMode: boolean) {
    this.authMode.next(authMode);
  }
}
