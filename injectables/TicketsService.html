import { Injectable } from '@angular/core';
import { BehaviorSubject, of } from 'rxjs';
import { switchMap, tap } from 'rxjs/operators';
import { listTicketsImpresora, TicketsImpresora } from '../../constants/tickets-list';

@Injectable({
  providedIn: 'root'
})
export class TicketsService {

  modoSeleccionarImpresora$ = new BehaviorSubject<boolean>(false);

  private updatelistaTicketsByMovimiento = new BehaviorSubject<TicketsImpresora[]>([]);
  updatelistaTicketsByMovimiento$ = this.updatelistaTicketsByMovimiento.asObservable();

  constructor() { }

  getTicketsLength() {
    return this.updatelistaTicketsByMovimiento.value?.length;
  }

  getTicketsByMovimiento() {
    return of(listTicketsImpresora).pipe(
      tap(list => {
        this.updatelistaTicketsByMovimiento.next([...list]);
      })
    );
  }

  addTicket(ticket: TicketsImpresora) {
    this.updatelistaTicketsByMovimiento.next([...this.updatelistaTicketsByMovimiento.value, ticket])
  }

  removeTicket(ticket: TicketsImpresora) {
    const newListState = this.updatelistaTicketsByMovimiento.value.filter((t) => t.impresora.id !== ticket.impresora.id)
    this.updatelistaTicketsByMovimiento.next([...newListState])
  }

  updateModoSeleccionarImpresora(value: boolean) {
    this.modoSeleccionarImpresora$.next(value ?? !this.modoSeleccionarImpresora$.value);
  }
}
