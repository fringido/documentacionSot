import { Injectable } from '@angular/core';
import { Apollo, gql, QueryRef } from 'apollo-angular';
import { BehaviorSubject, take, tap } from 'rxjs';
import {
  GetAllRoomsQuery,
  GetHabitacionIdQuery,
  Habitacion,
  HabitacionDetalladaQuery,
  HabitacionQuery,
} from '../../graphQL/serviciosGraph';

const GET_HABITACION = gql`
  query Habitacion($num: String!) {
    habitacion(id_o_numero: $num) {
      id
      numeroHabitacion
      tipoHabitacion {
        id
        nombre
      }
      estado
    }
  }
`;

const GET_HABITACION_ID = gql`
  query Habitacion($num: String!) {
    habitacion(id_o_numero: $num) {
      id
    }
  }
`;

const GET_HABITACION_detallada = gql`
  query Habitacion_Detallada($num: String!) {
    habitacion(id_o_numero: $num) {
      numeroHabitacion
      tipoHabitacion {
        nombre
      }
      estado
      rentas {
        id
        fechaRegistro
        fechaSalida
        fechaInicio
        fechaFin
        horasExtra
        hospedajesExtra
        personasExtra
        extras {
          fechaPago
          id
        }
        tarifa {
          nombre
          costoHabitacion
          horasExtraMax
          hospedajesExtraMax
          personasExtraMax
          costoPersonaExtra
          costoHospedajeExtra
          costoHoraExtra
        }
      }
    }
  }
`;

@Injectable({
  providedIn: 'root',
})
export class ObtenerHabitacionesService {
  habitacion!: Habitacion;
  idhabitacion!: GetHabitacionIdQuery['habitacion'];
  habitacionDetallada!: HabitacionDetalladaQuery['habitacion'];

  private habitacionSubjext = new BehaviorSubject<
    HabitacionQuery['habitacion']
  >(this.habitacion);
  habitacion$ = this.habitacionSubjext.asObservable();

  private habitacionIdSubjext = new BehaviorSubject<
    GetHabitacionIdQuery['habitacion']
  >(this.idhabitacion);
  habitacionId$ = this.habitacionIdSubjext.asObservable();

  private habitacionDetalladaSubjext = new BehaviorSubject<
    HabitacionDetalladaQuery['habitacion']
  >(this.habitacionDetallada);
  habitacionDetallada$ = this.habitacionDetalladaSubjext.asObservable();

  constructor(private apollo: Apollo) {}

  getHabitacio(id: string) {
    this.apollo
      .watchQuery<HabitacionQuery>({
        query: GET_HABITACION,
        variables: {
          num: id,
        },
      })
      .valueChanges.pipe(
        take(1),
        tap(({ data }) => {
          const { habitacion } = data;
          this.habitacionSubjext.next(habitacion);
        })
      )
      .subscribe();
  }

  getHabitacioId(id: string) {
    this.apollo
      .watchQuery<GetHabitacionIdQuery>({
        query: GET_HABITACION_ID,
        variables: {
          num: id,
        },
      })
      .valueChanges.pipe(
        take(1),
        tap(({ data }) => {
          const { habitacion } = data;
          this.habitacionIdSubjext.next(habitacion);
        })
      )
      .subscribe();
  }

  getHabitacioDetalle(id: string) {
    this.apollo
      .watchQuery<HabitacionDetalladaQuery>({
        query: GET_HABITACION_detallada,
        variables: {
          num: id,
        },
      })
      .valueChanges.pipe(
        take(1),
        tap(({ data }) => {
          const { habitacion } = data;
          this.habitacionDetalladaSubjext.next(habitacion);
        })
      )
      .subscribe();
  }
}
