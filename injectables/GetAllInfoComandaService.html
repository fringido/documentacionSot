const GET_ALL_INFO_COMANDA = gql`
  query GetAllInfoComandas {
    comandas {
      id
      fechaSolicitud
      fechaCancelacion
      fechaEntrega
      fechaPago
      renta {
        habitacion {
          numeroHabitacion
        }
      }

        cantidad

  providedIn: 'root',

export class GetAllInfoComandaService {
  queryRef!: QueryRef<any>;

  infoComanda!: GetAllComandasInfoQuery['comandas'];

    GetAllComandasInfoQuery['comandas']
  >(this.infoComanda);
  infoComanda$ = this.infoComandaSubjext.asObservable();

  constructor(
    private apollo: Apollo,

    ) {
      this.queryRef = this.apollo.watchQuery<any>({
        query: fakeQuery
      })
    }

    ngOnInit(){
      this.getArticculos()
      // this.getArticculosSuscri()
    }

  getArticculos() {
    return this.apollo
      .watchQuery<GetAllComandasInfoQuery>({
        query: GET_ALL_INFO_COMANDA,
      })
      .valueChanges.pipe(
        take(1),
        tap(({ data }) => {
          const { comandas } = data;
          this.infoComandaSubjext.next(comandas);
          console.log(comandas,'Cqr')

          return comandas
        })
      )
