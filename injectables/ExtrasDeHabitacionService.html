import { Injectable } from '@angular/core';
import { gql, Apollo } from 'apollo-angular';
import { BehaviorSubject, take, tap } from 'rxjs';
import { CreateExtraInput, GetExtrasQuery } from '../../graphQL/serviciosGraph';

const SALIDA_HABITACION = gql`
  mutation Liberate($id: ID!) {
    liberar_habitacion(id: $id)
  }
`;

const EXTRA_HABITACION = gql`
  mutation CrearExtras($extra: CreateExtraInput!) {
    crear_extra(createExtraInput: $extra)
  }
`;

const PAGAR_EXTRA = gql`
  mutation PagarExtra($id: ID!) {
    pagar_extra(id: $id)
  }
`;


const GET_EXTRAS = gql`
query GetExtras{
  extras{
    id
    renta {
      id
      habitacion {
        numeroHabitacion
      }
    }
    horas
    hospedajes
    personas
  }
}
`;

@Injectable({
  providedIn: 'root',
})
export class ExtrasDeHabitacionService {

  private extrasSubjext = new BehaviorSubject<GetExtrasQuery["extras"]>([])
  extras$ = this.extrasSubjext.asObservable();

  constructor(private apollo: Apollo) {

  }

  addExtraHabitacion(extra: CreateExtraInput) {
    this.apollo
      .mutate({
        mutation: EXTRA_HABITACION,
        variables: {
          extra,
        },
      })
      .subscribe(
        ({ data }: any): any => {
          console.log('Extra lista');
        },
        (error: any) => {
          console.log('there was an error sending the query', error);
        }
      );
  }

  salidaHabitacion(id: string) {
    this.apollo
      .mutate({
        mutation: SALIDA_HABITACION,
        variables: {
          id,
        },
      })
      .subscribe(
        ({ data }: any): any => {
          console.log('Salida lista');
        },
        (error: any) => {
          console.log('there was an error sending the query', error);
        }
      );
  }

  getextras() {
    this.apollo.watchQuery<GetExtrasQuery>({
      query: GET_EXTRAS
    }).valueChanges.pipe(
      take(1),
      tap(({ data }) => {
        const { extras } = data
        this.extrasSubjext.next(extras)
      })
    ).subscribe()
  }

  pagarExtra(id: string){
    this.apollo
    .mutate({
      mutation: PAGAR_EXTRA,
      variables: {
        id,
      },
    })
    .subscribe(
      ({ data }: any): any => {
        console.log('extra pagado');
      },
      (error: any) => {
        console.log('there was an error sending the query', error);
      }
    );
  }
}
