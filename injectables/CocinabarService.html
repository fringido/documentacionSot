import { Injectable } from '@angular/core';
import { gql } from 'apollo-angular';
import { iif, Observable, of } from 'rxjs';
import { mergeMap } from 'rxjs/operators';
import { ApiService } from 'src/app/core/services/api/api.service';
import { GetComandaService } from '../comanda/getComanda.service';

const GET_COMANDAS_COCINA = gql`
query Comandas {
  comandas {
    id
    fechaSolicitud
    fechaCancelacion
    fechaEntrega
    fechaPago
    renta {
      habitacion {
        numeroHabitacion
      }
    }
    alimentosComanda {
      cantidad
      comentario
      fechaPorEntregar
      cortesia
      alimento {
        nombre
        precio
      }
    }
  }
}
`;

const GET_COMANDAS_BAR = gql`
query Comandas {
  comandas {
    id
    fechaSolicitud
    fechaCancelacion
    fechaEntrega
    fechaPago
    renta {
      habitacion {
        numeroHabitacion
      }
    }
    tragosComanda {
      cantidad
      comentario
      fechaPorEntregar
      cortesia
      trago {
        nombre
        precio
      }
    }
  }
}
`;
@Injectable()
export class CocinabarService {

  constructor(
    private readonly apiService: ApiService,
    private readonly getComandaService: GetComandaService
  ) {
   }

  // obtenerDatosCocina u obtenerDatosBar dependiendo de la ruta elegida
  getCocinaBarData(filter: 'cocina' | 'bar'): Observable<any> {
    return filter === 'cocina' ?
      this.apiService.query({query: GET_COMANDAS_COCINA}) :
      this.apiService.query({query: GET_COMANDAS_BAR})
  }

  getCocinaBarSubs(filter: 'cocina' | 'bar'): Observable<any> {
    return filter === 'cocina' ? 
        this.getComandaService.comandasCocina$ : 
        this.getComandaService.comandasBar$
  }
}
