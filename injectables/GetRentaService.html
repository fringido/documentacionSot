import { Injectable } from '@angular/core';
import { Apollo, gql } from 'apollo-angular';
import { take, tap, BehaviorSubject } from 'rxjs';
import { RentaQuery } from '../../graphQL/serviciosGraph';
const GET_RENTA = gql `
query{
  rentas {
    habitacion {
      numeroHabitacion
      tipoHabitacion {
        nombre
      }
    }
    tarifa {
      nombre
      horasExtraMax
      hospedajesExtraMax
      personasExtraMax
      costoPersonaExtra
      costoHospedajeExtra
      costoHoraExtra
    }
    id
    horasExtra
    hospedajesExtra
    personasExtra
  }
}
`

@Injectable({
  providedIn: 'root'
})
export class GetRentaService {

  private rentasSubjext = new BehaviorSubject<RentaQuery["rentas"]>([])
  rentas$ = this.rentasSubjext.asObservable();

constructor(
  private apollo: Apollo
) { }

getRenta() {
  this.apollo.watchQuery<RentaQuery>({
    query: GET_RENTA
  }).valueChanges.pipe(
    take(1),
    tap(({ data }) => {
      const { rentas } = data
      this.rentasSubjext.next(rentas)
    })
  ).subscribe()
}

}
