import { Location } from '@angular/common';
import { Component, ElementRef, OnInit, QueryList, Renderer2, ViewChildren } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { RadioButton } from 'primeng/radiobutton';
import { debounceTime, distinctUntilChanged } from 'rxjs/operators';
import { ConfimModalMessageComponent } from 'src/app/core/components/confim-modal-message/confim-modal-message.component';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { impresorasList } from './constants/impresoras';

@Component({
  selector: 'app-lista',
  templateUrl: './lista.component.html',
  styleUrls: ['./lista.component.scss']
})
export class ListaComponent implements OnInit {

  visible = true;
  totalImpresoras!: any[];
  impresoras!: any[];

  @ViewChildren('printersRef') printersRef!: QueryList<ElementRef<HTMLDivElement>>;
  @ViewChildren('printersImg') printersImg!: QueryList<ElementRef<HTMLDivElement>>;
  @ViewChildren('pagesRef') pagesRef!: QueryList<ElementRef<RadioButton>>;

  selectedPrinterRefId?: string;
  selectedPrinterImg?: string;
  selectedPrinterId?: string;
  selectedPageId: number = 0;

  pages!: number;
  pageSelected = 0;

  filterText = new FormControl();

  constructor(
    private readonly location: Location,
    private readonly renderer: Renderer2,
    private readonly router: Router,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
    this.search();
  }

  salir() {
    this.location.back();
  }

  liberarVariables() {
    this.totalImpresoras = [...impresorasList];
    this.impresoras = this.totalImpresoras.slice(0, 8);
    this.pages = Math.ceil(this.totalImpresoras.length / 8);
  }

  search() {
    this.filterText = new FormControl("");
    this.filterText.valueChanges
      .pipe(debounceTime(400), distinctUntilChanged())
      .subscribe((v) => {
        this.unselectPrinters();
        if (v) {
          this.changePage(0);
          const filter = new RegExp(v, "i");
          const filterFields = ["nombre"];
          this.impresoras = this.totalImpresoras.filter((role) =>
            filterFields.some((field) => filter.test(role[field]))
          );
          this.pages = Math.ceil(this.impresoras.length / 8);
        } else {
          this.liberarVariables();
        }
    });
  }

  unselectPrinters() {
    this.printersRef.toArray().forEach((el, index) => {
      this.togglePrinterStatus({
        printerEl: el?.nativeElement,
        printerImgEl: this.printersImg.toArray()[index]?.nativeElement,
        fromClass: 'boxBadgeSelected',
        toClass: 'boxBadge',
        imgPath: '../../../../../assets/img/impresora.png'
      })
    });
    this.selectedPrinterRefId = undefined;
    this.selectedPrinterImg = undefined;
  }

  selectPrinter(printerRefId: string, printerImgRefId: string, printerId: string) {

    const getElement = (refs: QueryList<ElementRef>, id: string ) => {
      return refs.toArray().find(el => el.nativeElement.id === id)?.nativeElement;
    }

    const printer = getElement(this.printersRef, printerRefId);
    const prevPrinter = getElement(this.printersRef, this.selectedPrinterRefId!);


    const printerImg = getElement(this.printersImg, printerImgRefId);
    const prevPrinterImg = getElement(this.printersImg, this.selectedPrinterImg!);
    // Al cambiar de pagina se ejecuta este metodo, por lo tanto es posible que no se haya seleccionado
    // una impresora previamente, para evitar error property classList of undefined se añade esta validacion
    if(!printer || !printerImg ) {
      return;
    }
    if(printer === prevPrinter) {
      this.togglePrinterStatus({
        printerEl: prevPrinter,
        printerImgEl: prevPrinterImg,
        fromClass: 'boxBadgeSelected',
        toClass: 'boxBadge',
        imgPath: '../../../../../assets/img/impresora.png'
      });
      this.selectedPrinterRefId = undefined;
      this.selectedPrinterImg = undefined;
        return;
    }
    this.selectedPrinterId = printerId;
    if(prevPrinter && prevPrinterImg) {
      this.togglePrinterStatus({
        printerEl: prevPrinter,
        printerImgEl: prevPrinterImg,
        fromClass: 'boxBadgeSelected',
        toClass: 'boxBadge',
        imgPath: '../../../../../assets/img/impresora.png'
      });
    }
    this.selectedPrinterRefId = printerRefId;
    this.selectedPrinterImg = printerImgRefId;

    this.togglePrinterStatus({
      printerEl: printer,
      printerImgEl: printerImg,
      fromClass: 'boxBadge',
      toClass: 'boxBadgeSelected',
      imgPath: '../../../../../assets/img/impresora_seleccionado.png'
    });
  }

  private togglePrinterStatus(
    {printerEl, printerImgEl, fromClass, toClass, imgPath}:
    {printerEl: HTMLDivElement, printerImgEl: HTMLDivElement, fromClass: string, toClass: string, imgPath: string}
    ) {
    this.renderer.removeClass(printerEl, fromClass);
    this.renderer.addClass(printerEl, toClass);
    this.renderer.setAttribute(printerImgEl, 'src', imgPath)
  }

  /**
 * Método que se ejecuta al cambiar de página en la lista de impresoras, en caso de haber
 * una impresora seleccionada la deselecciona
 * @param {number} pageIndex
 * @returns void
 */
  changePage(pageIndex: number){
    
    if(this.selectedPageId === pageIndex) {
      return;
    }
    this.impresoras = this.totalImpresoras.slice(pageIndex * 8 , (pageIndex + 1) * 8);
    this.selectedPageId = pageIndex;
    this.selectPrinter(this.selectedPrinterRefId!, this.selectedPrinterImg!, this.selectedPrinterId!);
  }

  redirect(path: string) {
    this.router.navigateByUrl(path)
  }

  descartarImpresora() {
    // if(impresoraconfigurada) {
    // const ref = this.dialogService.open(MessageModalAutoclosableComponent, {
    //   data: {
    //     message: 'NO SE PUEDE DESCARTAR PORQUE TIENES UNA CONFIGURACIÓN PARA ESTA IMPRESORA'
    //   }
    // });
    // }
    const ref = this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: '¿DESEA DESCARTAR IMPRESORA?'
      }
    });

    ref.onClose.subscribe(({confirmed}) => {
      if(!confirmed) {
        return;
      }
      this.dialogService.open(MessageModalAutoclosableComponent, {
        data: {
          message: 'IMPRESORA DESCARTADA CON ÉXITO'
        }
      });
    });
  }

}
