import { AfterViewInit, Component, OnDestroy, OnInit, Renderer2 } from '@angular/core';
import { Location, CurrencyPipe } from '@angular/common';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { validaciones } from 'src/app/core/formsControl/validaciones';
import { SuscripcionEstadoHabitacionService } from 'src/app/core/services/habitaciones/suscripcion-estado-habitacion.service';
import { ViewChildren, QueryList, ElementRef } from '@angular/core';
import { DialogService } from 'primeng/dynamicdialog';
import { takeUntil } from 'rxjs/operators';
import { Subject } from 'rxjs';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { PaquetesList } from '../../constants/paquetes-list';
import { ValidacionesReactive } from 'src/app/core/formsControl/validaciones-reactive';

@Component({
  selector: 'app-editar',
  templateUrl: './editar.component.html',
  styleUrls: ['./editar.component.scss']
})
export class EditarComponent implements OnInit, OnDestroy {
  destroy$ = new Subject();
  visible = true;
  authMode = false;

  paqueteForm!: FormGroup;
  validaciones = new validaciones();
  validacionesReactive = new ValidacionesReactive();

  formChanged = false;
  tiposHabitacion!: string[];
  @ViewChildren('tiposHabRef') tiposHabRef!: QueryList<ElementRef<HTMLSpanElement>>;

  constructor(
    private readonly location: Location,
    private readonly fb: FormBuilder,
    private readonly renderer: Renderer2,
    private readonly estadoHabitacionService: SuscripcionEstadoHabitacionService,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.estadoHabitacionService.getTipos();
    this.iniciarVariables();
  }

  ngOnDestroy(): void {
    this.destroy$.next(true);
  }

  iniciarVariables() {
    
    this.paqueteForm = this.fb.group({
      name: [PaquetesList[0].name, [
        Validators.required,
        this.validacionesReactive.noExiste(PaquetesList.map(i => i.name), PaquetesList[0].name)
      ]],
      habitacion: [PaquetesList[0].roomType, Validators.required],
      price: [PaquetesList[0].price, Validators.required],
      discount: [PaquetesList[0].discount, Validators.required],
      cobroUnico: [PaquetesList[0].cobroUnico, Validators.required]
    });
    this.estadoHabitacionService.tipos$.pipe(takeUntil(this.destroy$)).subscribe(types => {
      if(!types) {
        return;
      }
      this.tiposHabitacion = types.map(type => type.nombre);
      this.tiposHabitacion[this.tiposHabitacion.length] = 'Todos';
      this.tiposHabitacion = [...this.tiposHabitacion]
      this.selectTipoHabitacion(types[0].nombre);
    });
    this.price?.valueChanges.subscribe(() => {
      this.discount?.disable({emitEvent: false});
      if(this.price?.value === 0 || this.price?.value === null) {
        this.price?.setValue(0, {emitEvent: false})
        this.discount?.enable({emitEvent: false});
      }
    });
    this.discount?.valueChanges.subscribe(() => {
      this.price?.disable({emitEvent: false});
      if(this.discount?.value === 0 || this.discount?.value === null) {
        this.discount?.setValue(0, {emitEvent: false})
        this.price?.enable({emitEvent: false});
      }
    });
  }

  get name() {
    return this.paqueteForm.get('name')
  }
  get habitacion() {
    return this.paqueteForm.get('habitacion')
  }
  get price() {
    return this.paqueteForm.get('price')
  }
  get discount() {
    return this.paqueteForm.get('discount')
  }
  get cobroUnico() {
    return this.paqueteForm.get('cobroUnico')
  }

  selectTipoHabitacion(tipoHabitacion: string) {
    
    this.habitacion?.setValue(tipoHabitacion);
    
    if(tipoHabitacion === 'Todos') {
      return this.tiposHabRef?.toArray().forEach((el) => {
        this.renderer.addClass(el.nativeElement, 'boxBadgeSelected');
      })
    }
    const tipoHabSelected = this.tiposHabRef?.toArray().find((el) => el.nativeElement.id === 'tipo_' + tipoHabitacion);
    const todosOption = this.tiposHabRef?.toArray().find((el) => el.nativeElement.id === 'tipo_Todos');
    tipoHabSelected?.nativeElement.classList.toggle('boxBadgeSelected');
    todosOption?.nativeElement.classList.remove('boxBadgeSelected');
  }
  
  actualizarPaquete() {
    console.log({...this.paqueteForm.value, name: this.name?.value?.toUpperCase()});
    this.authMode = true;
  }

  actualizarPaqueteAutenticado(confirmed: boolean) {
    this.authMode = false;
    if(!confirmed) {
      return;
    }
    const ref = this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: `PAQUETE "${this.name?.value.toUpperCase()}" MODIFICADO CON ÉXITO`
      }
    })
    ref.onClose.subscribe(() => {
      this.salir();
    })
  }

  salir() {
    this.location.back();
  }

}
