import { Component, OnInit } from '@angular/core';
import { Location, CurrencyPipe } from '@angular/common';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Observable } from 'rxjs';
import { debounceTime } from 'rxjs/operators';
import { validaciones } from 'src/app/core/formsControl/validaciones';
import { TiposHabitacionsesQuery } from 'src/app/core/graphQL/serviciosGraph';
import { SuscripcionEstadoHabitacionService } from 'src/app/core/services/habitaciones/suscripcion-estado-habitacion.service';
import { DialogService } from 'primeng/dynamicdialog';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { PaquetesList } from '../../constants/paquetes-list';

@Component({
  selector: 'app-editar',
  templateUrl: './editar.component.html',
  styleUrls: ['./editar.component.scss']
})
export class EditarComponent implements OnInit {
  visible = true;
  authMode = false;

  paqueteForm!: FormGroup;
  validaciones = new validaciones();

  formChanged = false;
  tiposHabitacion$!: Observable<TiposHabitacionsesQuery['tipos_habitaciones']>

  constructor(
    private readonly location: Location,
    private readonly fb: FormBuilder,
    private readonly estadoHabitacionService: SuscripcionEstadoHabitacionService,
    private readonly currencyPipe: CurrencyPipe,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.iniciarVariables();
    this.estadoHabitacionService.getTipos();
  }

  iniciarVariables() {
    this.tiposHabitacion$ = this.estadoHabitacionService.tipos$;
    
    this.paqueteForm = this.fb.group({
      name: [PaquetesList[0].name, Validators.required],
      habitacion: [PaquetesList[0].roomType, Validators.required],
      price: [this.currencyPipe.transform(PaquetesList[0].price), Validators.required],
      discount: [this.currencyPipe.transform(PaquetesList[0].discount), Validators.required],
      cobroUnico: [PaquetesList[0].cobroUnico, Validators.required]
    });
    this.price?.valueChanges.pipe(
      debounceTime(1000)
    ).subscribe(v => {
      this.setInputFormat(this.price!, v);
    });
    this.discount?.valueChanges.pipe(
      debounceTime(1000)
    ).subscribe(v => {
      this.setInputFormat(this.discount!, v);
    });
    this.paqueteForm.valueChanges.subscribe(() => this.formChanged = true);
  }

  setInputFormat(formControl: AbstractControl<any>, v: string) {
    formControl.setValue(this.currencyPipe.transform(this.currencyInputChanged(v)), {emitEvent: false})
  }

  currencyInputChanged(value: string) {
    const num = Number(value.replace(/[$,]/g, ""));
    if(num < 1) {
      return num * 1000
    }
    return num
  }

  get name() {
    return this.paqueteForm.get('name')
  }
  get habitacion() {
    return this.paqueteForm.get('habitacion')
  }
  get price() {
    return this.paqueteForm.get('price')
  }
  get discount() {
    return this.paqueteForm.get('discount')
  }
  get cobroUnico() {
    return this.paqueteForm.get('cobroUnico')
  }

  actualizarPaquete() {
    console.log({...this.paqueteForm.value, name: this.name?.value.toUpperCase(), price: this.currencyInputChanged(this.price?.value), discount: this.currencyInputChanged(this.discount?.value)});
    this.authMode = true;
  }

  actualizarPaqueteAutenticado(confirmed: boolean) {
    this.authMode = false;
    if(!confirmed) {
      return;
    }
    const ref = this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: `PAQUETE "${this.name?.value.toUpperCase()}" MODIFICADO CON ÉXITO`
      }
    })
    ref.onClose.subscribe(() => {
      this.salir();
    })
  }

  salir() {
    this.location.back();
  }

}
