  formCreate() {
    this.form = this.fb.group({
      fechaInicio: [''],
      fechaFinal: [''],
      turno: [''],
      personal: [''],
      habitacion: [''],
    });

    var res1 = new Date();
    var res2 = new Date();
    res1.setDate(res1.getDate() + 1);
    res2.setDate(res2.getDate() - 1);
    this.form.get('fechaInicio')?.patchValue(res2)
    this.form.get('fechaFinal')?.patchValue(res1)
    this.limpieza = this.limpiezaAll.filter(limpieza => new Date(limpieza.inicioLimpieza) >= res2 )
    this.limpieza = this.limpieza.filter(limpieza => new Date(limpieza.finLimpieza) <= res1)

    this.form.valueChanges
    .pipe(debounceTime(400), distinctUntilChanged())
    .subscribe((v) => {
      if (v) {
        this.limpieza = this.limpiezaAll.filter(limpieza => new Date(limpieza.inicioLimpieza) >= new Date(v.fechaInicio) || v.fechaInicio == '')
        this.limpieza = this.limpieza.filter(limpieza => new Date(limpieza.finLimpieza) <= new Date(v.fechaFinal) || v.fechaFinal == '')
        this.limpieza = this.limpieza.filter(limpieza => limpieza.recamerera || limpieza.supervisor == v.personal.nombre || v.nombre == '')
      } else {
        this.limpieza = this.limpiezaAll;
      }
    });
  }
