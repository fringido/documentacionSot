 * Tarjeta de el componente de {@link LiberarPagarRoomServiceComponent}

@Component({
  selector: 'app-tarjetas-comanda',
  templateUrl: './tarjetas-comanda.component.html',
  styleUrls: ['./tarjetas-comanda.component.scss']
})
export class TarjetasComandaComponent {

  @Input() comanda: any
  @Input() numHabitacion!: string;
  authMode = false
  endDate!: Date;
  cancelarComandaBool= false
  formCancelarComanda!: FormGroup

  countdown: any;
  timeLeft: any;
  timeDiff!: number
  datosComanda!: { comanda: any; accion: string; }

  constructor(
    private accionesComandaService: ComandaPagoEnvioService,
    private pagoService: ComunicacionPagoService,
    private router: Router,
    private fb: FormBuilder,
    private cancelarComandaService: CancelarComandaService,
    public dialogService: DialogService,

  ) {
  }


  ngOnInit(): void {
    this.endDate = new Date(this.comanda.fechaSolicitud);
    this.countdown = setInterval(() => {
      this.timeDiff = new Date().getTime() -  this.endDate.getTime()
      if (this.timeDiff <= 0) {
        } else {
        const hours = Math.floor((this.timeDiff / (1000 * 60 * 60)) % 24);
        const minutes = Math.floor((this.timeDiff / 1000 / 60) % 60);
        const seconds = Math.floor((this.timeDiff / 1000) % 60);
        this.timeLeft = `${hours}h ${minutes}m ${seconds}s`;
      }
    }, 1000);
    this.formCancelarComanda = this.fb.group({
      motivo:['',[Validators.required]]
    });
  }
/**
 * Al dar aceptar en el boton del componente individual dependiendo del estado en el que este hace una accion
 * si el estado es por cobrar envia una mutacion de entrega de comanda
 * se es Por pagar re direcciona a el pago de la comanda
 * @param comanda al dar aceptar se envia la comanda a la funsion
 */
  aceptar(comanda: any) {
    if (comanda.estadoPedido == 'Por cobrar') {
      this.accionesComandaService.entregarComanda(comanda.id).subscribe(
        ({ data }:any):any => {
          let mensaje = 'COMANDA PAGADA'
          this.mensaje(mensaje)
        },
        (error: any) => {
          this.mensaje(error.message)
        }
      )

      this.authMode = false;
    }
    if (comanda.estadoPedido == 'Por pagar') {
      this.pagoService.infoPago$ = comanda.id
      this.pagoService.pago$ = comanda.total
      this.router.navigate([`/hotel/roomService/pago/${this.numHabitacion}`]);
    }
  }

/**
 * Funcion para lansar modal de mensajes
 * @param mensaje Mensaje que llegue
 */
  mensaje(mensaje: string) {
    this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: mensaje
      }
    })
  }

/**
 * Componente para cancelar comanda
 * Si fue correcta la autentificacion envia la mutacion de cancelar comanda
 * @param event Dependiendo de lo que el componente de autentificacion regrese hace una accion
 * @returns nada
 */
  cancelar(event: boolean) {
    if (!event) {
      this.authMode = false;
      return;
    }
    this.cancelarComandaService.cancelarComandaa(this.comanda.id, this.formCancelarComanda.controls['motivo'].value);
    this.cancelarComandaBool = false

    this.cancelarComandaService.mensaje$.subscribe(mensaje =>{
      this.mensaje(mensaje)
      this.authMode = false;
    })

  }

   //* FUNCION PARA LOS CAMBIOS DE ESTILO
/**
 * regresa el estilo para dar color a las comandas
 * @param estado Recibe el estado de la comanda
 * @returns Estilo que esta en scss
 */
  style(estado: string) {

    if (estado == 'Preparación') {
      return 'preparacion'
    }
    else if (estado == 'Por cobrar') {
      return 'porcobrar'
    }
    else if (estado == 'Por pagar') {
      return 'porpagar'
    }
    return ''
  }
  //* ------------------------------------------

}
