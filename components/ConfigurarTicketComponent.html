import { Component, OnDestroy, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { Router } from '@angular/router';
import { TicketsService } from '../../services/tickets/tickets.service';
import { TicketsImpresora } from '../../constants/tickets-list';
import { switchMap, tap, takeUntil } from 'rxjs/operators';
import { of, Subject } from 'rxjs';
import { DialogService } from 'primeng/dynamicdialog';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { listaMovimientos } from '../../constants/movimientos-list';


@Component({
  selector: 'app-configurar-ticket',
  templateUrl: './configurar-ticket.component.html',
  styleUrls: ['./configurar-ticket.component.scss']
})
export class ConfigurarTicketComponent implements OnInit, OnDestroy {

  private readonly destroy$ = new Subject<boolean>();

  visible = true;
  modoSelectImpresora$ = this.ticketsService.modoSeleccionarImpresora$;
  ticketsListByType!: TicketsImpresora[];

  ticketsChanged = false;

  movimientoSeleccionado!: any;

  constructor(
    private readonly location: Location,
    private readonly router: Router,
    private readonly ticketsService: TicketsService,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    of(listaMovimientos[0]).pipe(
      takeUntil(this.destroy$)
    ).subscribe(mov => this.movimientoSeleccionado = mov);
    this.ticketsService.getTicketsByMovimiento().pipe(
      takeUntil(this.destroy$),
      tap((ticketsListByType) => {
        if(!this.ticketsService.getTicketsLength()){
          this.visible = false;
          this.toggleModoSelectImpresora(true);
          this.ticketsListByType = [...ticketsListByType];
        }
      }),
      switchMap(() => this.ticketsService.updatelistaTicketsByMovimiento$)
    ).subscribe(ticketsListByType => {
      this.ticketsListByType = [...ticketsListByType];
    });

  }

  ngOnDestroy(): void {
    this.destroy$.next(true);
    this.destroy$.unsubscribe();
  }

  toggleModoSelectImpresora(value : boolean) {
    this.ticketsService.updateModoSeleccionarImpresora(value);
  }

  guardar() {
    console.log({ticketsChanged: this.ticketsChanged});

    const ref = this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: `MOVIMIENTO ${this.movimientoSeleccionado.nombre} CONFIGURADA CON ÉXITO`
      }
    });
    ref.onClose.subscribe(() => {
      this.salir();
    })
  }

  salir() {
    this.location.back();
  }

}
