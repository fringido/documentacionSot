import { Component, OnDestroy, OnInit } from '@angular/core';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { MatriculaService } from '../../services/matricula/matricula.service';

@Component({
  selector: 'app-detalle',
  templateUrl: './detalle.component.html',
  styleUrls: ['./detalle.component.scss']
})
export class DetalleComponent implements OnInit, OnDestroy {

  destroy$: Subject<boolean> = new Subject<boolean>();
  showReporteDetalle!: boolean;
  // TODO: tipar reportematricula
  reporteMatricula!: any;

  constructor(
    private readonly matriculaService: MatriculaService
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
  }

  ngOnDestroy(): void {
    this.matriculaService.updateMatriculaDetails(null);
    this.destroy$.next(true);
    this.destroy$.unsubscribe();
  }
  
  liberarVariables() {
    this.matriculaService.matriculaDetails$
    .pipe(takeUntil(this.destroy$))
    .subscribe((matricula) => {
      if(matricula?.fechaCreacion) {
        this.showReporteDetalle = true
      }
      this.reporteMatricula = matricula;
    });

  }

}
