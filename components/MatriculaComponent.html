import { Location } from '@angular/common';
import { Component, OnDestroy, OnInit } from '@angular/core';
import { DialogService } from 'primeng/dynamicdialog';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { MatriculaService } from './services/matricula/matricula.service';

@Component({
  selector: 'app-matricula',
  templateUrl: './matricula.component.html',
  styleUrls: ['./matricula.component.scss']
})
export class MatriculaComponent implements OnInit, OnDestroy {

  destroy$: Subject<boolean> = new Subject<boolean>();

  nuevoReporte!: {matricula: string, descripcion: string};

  visible = true;
  validBotton!: boolean;

  authMode!: boolean;

  constructor(
    private readonly location: Location,
    private readonly matriculaService: MatriculaService,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
  }

  liberarVariables() {
    this.matriculaService.nuevoReporteValido$
    .pipe(takeUntil(this.destroy$))
    .subscribe((esValido) => {
      this.validBotton = esValido;
    });
    this.matriculaService.authMode$
    .pipe(takeUntil(this.destroy$))
    .subscribe(isAuthMode => {
      this.authMode = isAuthMode;
    });
  }

  ngOnDestroy(): void {
    this.destroy$.next(true);
    this.destroy$.unsubscribe();
  }

  crearReporte() {

    console.log({reporte: this.nuevoReporte});

    this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: 'Reporte creado con éxito'
      }
    });
    this.matriculaService.updateMatricula('');
  }

  salir() {
    this.location.back();
  }

  confirmarInhabilitarMatricula(event: boolean) {
    if(!event) {
      return this.matriculaService.updateAuthMode(false);
    }
    this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: 'MATRÍCULA INHABILITADA CON ÉXITO'
      }
    });
    return this.matriculaService.updateAuthMode(false);
  }

}
