import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { ConfimModalMessageComponent } from 'src/app/core/components/confim-modal-message/confim-modal-message.component';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { HabitacionQuery } from 'src/app/core/graphQL/serviciosGraph';
import { MutarEstadoHabitacionService } from 'src/app/core/services/habitaciones/mutar-estado-habitacion.service';
import { ObtenerHabitacionesService } from 'src/app/core/services/habitaciones/obtener-habitaciones.service';
import { RoomStatusEnum } from 'src/app/room-types/enums/room-status.enum';
import { RoomService } from 'src/app/room-types/services/room/room.service';

@Component({
  selector: 'app-cancelar-entrada',
  templateUrl: './cancelar-entrada.component.html',
  styleUrls: ['./cancelar-entrada.component.scss'],
})
export class CancelarEntradaComponent implements OnInit {
  authMode = false;
  form!: FormGroup;
  habitacion!: HabitacionQuery['habitacion'];
  habitacion$ = this.habitacionService.habitacion$;
  numHabitacion: string;

  constructor(
    private readonly router: Router,
    public dialogService: DialogService,
    private route: ActivatedRoute,
    private readonly roomService: RoomService,
    private fb: FormBuilder,
    private habitacionService: ObtenerHabitacionesService,
    private mutarEstado: MutarEstadoHabitacionService
  ) {
    this.numHabitacion = String(this.route.snapshot.paramMap.get('roomNumber'));
    this.getInfo(this.numHabitacion);
  }

  ngOnInit(): void {
    this.mensajes();
  }

  mensajes() {
    const ref = this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message:
          '¿Desea cancelar la entrada de la habitación ' +
          this.numHabitacion +
          '?',
      },
    });
    ref.onClose.subscribe(({ confirmed }) => {
      if (!confirmed) {
        this.salir();
        return
      }
      this.aceptar();
    });
  }

  getInfo(numHabitacion: string) {
    this.habitacionService.getHabitacio(numHabitacion);
    this.habitacion$.subscribe((data) => {
      this.habitacion = data;
    });
  }

  aceptar() {
    const estado = RoomStatusEnum.LIBRE;
    this.mutarEstado.cambiarEstado(this.habitacion.id, estado);
    this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: 'ENTRADA CANCELADA',
      },
    });
    this.salir();
  }

  salir() {
    this.router.navigate(['hotel']);
  }
}
