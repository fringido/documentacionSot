import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { DialogService } from 'primeng/dynamicdialog';
import { ActivatedRoute, Router } from '@angular/router';
import { Location } from '@angular/common';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';

import { ObtenerHabitacionesService } from 'src/app/core/services/habitaciones/obtener-habitaciones.service';
import { MutarEstadoHabitacionService } from 'src/app/core/services/habitaciones/mutar-estado-habitacion.service';
import { HabitacionQuery } from 'src/app/core/graphQL/serviciosGraph';
import { ComunicacionPagoService } from '../../../entrada-habitacion/services/comunicacionPago.service';
import { ComandaPagoEnvioService } from 'src/app/cocinabar/services/comanda/comandaPagoEnvio.service';


@Component({
  selector: 'app-pago-comandas',

  styleUrls: ['./pago-comandas.component.scss']
})


export class PagoComandasComponent implements OnInit {

  /**
 * Inicio de variables
 */
  //* Inicio de variables
  display = true
  total= this.pagoService.pago$
  form!: FormGroup
  tipoDePago = new FormControl(null);

  tiposDePago = [
    { name: 'Efectivo', value: 1 },
    { name: 'Tarjeta', value: 2 },
    { name: 'Mixto', value: 3 },
    { name: 'Cortesía', value: 4 },
    { name: 'Consumo Inteerno', value: 5 }
  ];
  //* -------------------------------

habitacion!: HabitacionQuery["habitacion"]
pago : number = 0
/**
 * Constructor
 */
  constructor(
    private fb: FormBuilder,
    private location: Location,
    public dialogService: DialogService,
    private router: Router,
    private pagoService: ComunicacionPagoService,
    private comandaPagoService: ComandaPagoEnvioService
  ) {
    this.formoCreate();
  }
/**
 * @ignore
 */
  ngOnInit(): void {
    this.pago = this.pagoService.pago$
    this.menu();
  }


  //* Observa los cambios del menu para mostrar el formulario y crear el FormGroup
  /**
 * Observa los cambios del menu para mostrar el formulario y crear el FormGroup
 */
  menu() {
    this.tipoDePago.valueChanges.subscribe((d) => {
      if (d == 1) {
        this.form = this.fb.group({
          tipoDePago: [this.tipoDePago.value],
          total: [this.total]
        })
      }
      if (d == 4) {
        this.form = this.fb.group({
          tipoDePago: [this.tipoDePago.value],
          total: [this.total],
          observacion: ['', Validators.required]
        })
      }
      if (d == 5) {
        this.form = this.fb.group({
          tipoDePago: [this.tipoDePago.value],
          total: [this.total],
          empleado: ['', Validators.required],
          observacion: ['', Validators.required],
        })
      }

    })
  }
  //* ------------------------------------------------------

  //* Trae el de pago mixto y de tarjeta
 /**
 * Trae el de pago mixto y de tarjeta
 */
  addForm(form: FormGroup) {
    this.form = form
  }
  //* ------------------------------------------------------

  //* Inicia el formulario en nada
/**
 * Inicia el formulario en nada
 */
  formoCreate() {

    this.form = this.fb.group({
    })
  }
  //* ------------------------------------------------------


  //* BOtones de modal
    /**
 * Botones de modal
 */
  aceptar() {
  const id = String(this.pagoService.infoPago$)
    const genDialogMessage = (message: string) => {
      return this.dialogService.open(MessageModalAutoclosableComponent, {data: { message }});
    };
    const ref = genDialogMessage('PAGO REALIZADO CON ÉXITO')
    this.comandaPagoService.pagarComanda(id)
    this.router.navigate([`/hotel`]);
  }
  /**
 * Boton para salir
 */
  salir() {
    this.location.back()
  }
  //* ------------------------------------------------------


}
