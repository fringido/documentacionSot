import { Component, ElementRef, OnInit, Renderer2, ViewChild } from '@angular/core';
import { printStyle } from './constants/print-style';

@Component({
  selector: 'app-ticket-renta',
  templateUrl: './ticket-renta.component.html',
  styleUrls: ['./ticket-renta.component.scss']
})
export class TicketRentaComponent implements OnInit {

  @ViewChild('ticketHabitacion') ticketHabitacion!: ElementRef<HTMLHtmlElement>;

  rentaHabitacion = {
    idEmpleado: 0
  }

  baseUrl = location.origin

  constructor(
    private readonly renderer: Renderer2
  ) { }

  ngOnInit(): void {

  }

  imprSelec() {
    const style = this.renderer.createElement('style');
    style.innerHTML = printStyle;
    this.renderer.appendChild(this.ticketHabitacion.nativeElement, style);
    const ventimp = window.open('', 'popimpr')!;
    ventimp.document.write(this.ticketHabitacion.nativeElement.innerHTML);
    
    const onClose = () => {
      ventimp.document.close();
      ventimp.close();
      window.close();
    }
    setTimeout(() => {ventimp.print(); onClose()}, 100);
    ventimp.onclose = onClose
  }

}
