import { AfterViewInit, Component, HostListener, Input, OnDestroy, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { ModalService } from 'src/app/core/services/modal.service';
import { SidebarService } from '../../../sidebar/services/sidebar/sidebar.service';
import { RoomStatusEnum } from '../../enums/room-status.enum';
import { RoomService } from '../../services/room/room.service';
import { ConfimModalMessageComponent } from '../../../core/components/confim-modal-message/confim-modal-message.component';
import { MutarEstadoHabitacionService } from 'src/app/core/services/habitaciones/mutar-estado-habitacion.service';
import { PrintService } from 'src/app/core/services/print/print.service';
import {  HabitacionDetalladaQuery } from 'src/app/core/graphQL/serviciosGraph';
import { Room } from '../../interfaces/room.interface';
import { SuscripcionEstadoHabitacionService } from 'src/app/core/services/habitaciones/suscripcion-estado-habitacion.service';

@Component({
  selector: 'app-room-details',
  templateUrl: './room-details.component.html',
  styleUrls: ['./room-details.component.scss']
})
export class RoomDetailsComponent implements OnInit, OnDestroy,AfterViewInit {

  destroy$: Subject<boolean> = new Subject<boolean>();

  authMode = {
    cancelarBloqueo: false
  }

  readonly AvailableRoomStatus = RoomStatusEnum;
  display: boolean = false;
  public isCollapsed = {
    data: false,
    details: false,
    roomService: false
  };

  // Variables provisionales la habitacion seleccionada se va a obtener por medio de los params en la ruta
  selectedRoom!: Room ;

  isModoCambioHabitacion!: any;

  isModalOpen!: boolean;

  // habitacionDetallada$ = this.optenerHAbitacion.habitacionDetallada$

  habitacion!: HabitacionDetalladaQuery["habitacion"];

  habitaciones$ = this.havitacionService.habitaciones$

  ultimaRenta!:number
undefinide: any;

  constructor(
    private readonly router: Router,
    public dialogService: DialogService,
    private modalService: ModalService,
    private readonly sidebarService: SidebarService,
    private readonly roomService: RoomService,
    private readonly printService: PrintService,
    private mutarEstado: MutarEstadoHabitacionService,
    private havitacionService: SuscripcionEstadoHabitacionService
  ) {
  }
  ngOnDestroy(): void {
    this.destroy$.next(true);
    this.destroy$.unsubscribe();
    this.havitacionService.getHabitaciones();
  }

  ngAfterViewInit(){
  }

  ngOnInit(): void {
    this.inicioVariables()
  }

  inicioVariables() {

    this.roomService.selectedRoom$
    .pipe(takeUntil(this.destroy$))
    .subscribe((room) => {
      this.selectedRoom = room
      this.habitaciones$.pipe(takeUntil(this.destroy$)).subscribe(data => {
        const habitacion = data.filter( data => data.numeroHabitacion == this.selectedRoom.roomNumber)
        this.habitacion = habitacion[0]

      })
    });
    this.roomService.modoAppHabitacion$
    .pipe(takeUntil(this.destroy$))
    .subscribe((state) => {
      this.isModoCambioHabitacion = state.cambio;
    });
    this.modalService.isModalOpen$
    .pipe(takeUntil(this.destroy$))
    .subscribe((isOpen) => this.isModalOpen = isOpen)
  }


  @HostListener('document:keydown', ['$event'])
  onEscapeHandler(event: KeyboardEvent) {
    if (event.key !== 'Escape' || this.isModalOpen) {
      return;
    }
    // si se está mostrando una ruta en forma de modal se cancela la deselección de la habitacion
    this.exit();

  }

  mensaje(mensaje: string) {
    return this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: mensaje
      },
    });
  }

  mensajeConfirmacion(mensaje:string) {
    return this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: mensaje
      }
    });
  }

  exit() {
    if (this.isModoCambioHabitacion) {
      return this.togggleModoCambioHabitacion(false);
    }
    if (this.router.routerState.snapshot.url === '/hotel') {
      this.sidebarService.setSidebarState('home')
    }
  }

  entrada() {
    const id = this.selectedRoom.id;
    console.log(id)
    const estado = 'Preparada'
    this.mutarEstado.cambiarEstado(id, estado)
    const mensaje = 'SE HA DADO ENTRADA DE FORMA EXITOSA'
    this.mensaje(mensaje)


  }

  cancelarLimpieza() {
    const id = this.selectedRoom.id;
    console.log(id)
    const estado = 'Sucia'
    const ref = this.mensajeConfirmacion(`¿Estás seguro que quieres cancelar la limpieza de la habitación ${this.selectedRoom.roomNumber}?`);
    ref.onClose.subscribe(({ confirmed }) => {
      if (!confirmed) {
        return;
      }
      const mensaje = 'LIMPIEZA CANCELADA CON ÉXITO'

      this.mutarEstado.cambiarEstado(id, estado)
      this.mensaje(mensaje);

    });
  }

  confirmarReservacion() {
    const id = this.selectedRoom.id
    console.log(id)
    const estado = 'Reservada/Preparada'
    const mensaje = 'SE HA CONFIRMADO LA RESERVACIÓN'
    this.mutarEstado.cambiarEstado(id, estado)
    this.mensaje(mensaje)
    // this.roomService.updateSelectedRoom({
    //   ...this.selectedRoom,
    //   status: RoomStatusEnum.RESERVADA_PREPARADA
    // });
  }


  salida() {
    const ref = this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: `Estas por liberar la habitación ${this.selectedRoom.roomNumber}`,
      },
    });
    ref.onClose.subscribe(({ confirmed }) => {
      if (!confirmed) {
        return;
      }
      this.router.navigate([`hotel/salida/confirmar/${this.selectedRoom.roomNumber}`])
    });
  }

  togggleModoCambioHabitacion(active?: boolean) {
    this.roomService.updateModoAppHabitacion({ cambio: active ?? !this.isModoCambioHabitacion });
  }

  openModal(rute: string) {
    this.router.navigate([`hotel/${rute}/${this.selectedRoom.roomNumber}`])
  }

  cancelarBloqueo() {
    const ref = this.mensajeConfirmacion('¿ESTÁS SEGURO DE QUERER CANCELAR EL DESHABILITADO?')

    ref.onClose.subscribe(({ confirmed }) => {
      if (!confirmed) {
        return;
      }
      this.authMode.cancelarBloqueo = true;
    });
  }

  cancelarBloqueoAutenticado() {
    const message = 'SE HA CANCELADO EL DESHABILITADO'
    const estado = RoomStatusEnum.LIBRE

    this.mensaje(message)

    this.authMode.cancelarBloqueo = false;
    this.mutarEstado.cambiarEstado(this.selectedRoom.id, estado)

  }

  imprimirTicket() {
    this.printService.open(['ImprimirTicketHabitacion'])
  }

  abrirCamaristasEnTurno() {
    this.router.navigate(['hotel', 'camaristasEnTurno']);
  }

  reportarMatricula() {
    this.router.navigate(['hotel', 'reportar', 'matricula']);
  }


}
