import { CommonModule } from '@angular/common';
import { Component, EventEmitter, Input, OnDestroy, OnInit, Output } from '@angular/core';
import { FormControl, ReactiveFormsModule } from '@angular/forms';
import { CurrencyMaskModule } from 'ng2-currency-mask';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { validaciones } from 'src/app/core/formsControl/validaciones';

@Component({
  selector: 'app-card-extra',
  templateUrl: './card-extra.component.html',
  styleUrls: ['./card-extra.component.scss'],
  standalone: true,
  imports: [CurrencyMaskModule, CommonModule, ReactiveFormsModule]
})
export class CardExtraComponent implements OnInit, OnDestroy {

  private readonly destroy$ = new Subject<boolean>()

  @Input('title') title!: string;
  @Input('isCurrency') isCurrency!: boolean;
  @Input('value') value!: number;
  @Input('disabled') disabled: boolean = false;

  @Output('valueChange') valueChange = new EventEmitter<number>();

  inputValue = new FormControl(0);


  validaciones = new validaciones();

  constructor() { }
  ngOnDestroy(): void {
    this.destroy$.next(true);
  }


  ngOnInit(): void {

    if(this.disabled) {
      this.inputValue.disable();
    }

    this.inputValue.valueChanges
    .pipe(takeUntil(this.destroy$))
    .subscribe((v) => this.valueChange.emit(v ?? 0));
  }

}
