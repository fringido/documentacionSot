import { Location } from '@angular/common';
import { Component, OnInit, Renderer2, ViewChildren, QueryList, ElementRef } from '@angular/core';
import { FormBuilder, Validators } from '@angular/forms';
import { DialogService } from 'primeng/dynamicdialog';
import { ConfimModalMessageComponent } from 'src/app/core/components/confim-modal-message/confim-modal-message.component';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { validaciones } from 'src/app/core/formsControl/validaciones';

@Component({
  selector: 'app-registro',
  templateUrl: './registro.component.html',
  styleUrls: ['./registro.component.scss']
})
export class RegistroComponent implements OnInit {

  authMode = false;

  visible = true;
  validaciones = new validaciones();
  @ViewChildren('diasRef') diasRef!: QueryList<ElementRef<HTMLSpanElement>>;

  page = 0;
  extrasOptions = [
    {title: 'Habitación', isCurrency: true, value: 0},
    {title: 'Horas Extra', isCurrency: true, value: 0},
    {title: 'Persona Extra', isCurrency: true, value: 0},
    {title: 'Hospedaje Extra', isCurrency: true, value: 0},
    {title: 'Lím. Persona Extra', isCurrency: false, value: 0},
    {title: 'Lím. Horas Extra', isCurrency: false, value: 0},
    {title: 'Lím. Hospedaje Extra', isCurrency: false, value: 0},
  ]

  readonly diasOptions = [
    'Lunes',
    'Martes',
    'Miércoles',
    'Jueves',
    'Viernes',
    'Sábado',
    'Domingo',
    'Todos',
  ]

  tarifaForm = this.fb.group({
    name: ['', Validators.required],
    duracion: ['', Validators.required],
    horario_inicio: ['', Validators.required],
    horario_fin: ['', Validators.required],
    fecha_inicio: ['', Validators.required],
    fecha_fin: ['', Validators.required],
    tipo_hab: [0, Validators.required],
  })

  extrasOptionsToShow: typeof this.extrasOptions = []

  constructor(
    private readonly location: Location,
    private readonly renderer: Renderer2,
    private readonly fb: FormBuilder,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.extrasOptionsToShow = this.extrasOptions.slice(0, 4);
  }

  salir() {
    this.location.back();
  }

  changePage({page}: {page: number}) {
    this.extrasOptionsToShow = this.extrasOptions.slice((page * 4), (page  * 4) + 4);
  }

  onExtraChange(extra: {title: string, isCurrency: boolean, value: number}, event: number) {
    const extraToChangeIndex = this.extrasOptions.findIndex((opt) => opt.title === extra.title);
    this.extrasOptions[extraToChangeIndex].value = event;
  }

  selectDia(dia: string) {
    if(dia === 'Todos') {
      return this.diasRef.toArray().forEach((el) => {
        this.renderer.addClass(el.nativeElement, 'boxBadgeSelected');
      })
    }

    const diaSelected = this.diasRef.toArray().find((el) => el.nativeElement.id === 'dia_' + dia);
    const todosOption = this.diasRef.toArray().find((el) => el.nativeElement.id === 'dia_Todos');

    todosOption?.nativeElement.classList.remove('boxBadgeSelected');
    diaSelected?.nativeElement.classList.toggle('boxBadgeSelected');
  }

  registro() {
    const ref = this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: `SE CREARÁ LA TARIFA CON EL NOMBRE ${this.tarifaForm.get('name')?.value}`
      }
    });

    ref.onClose.subscribe(({confirmed}) => {
      if(!confirmed) {
        return;
      }
      this.authMode = true;
    })
  }

  registroAutenticado(confirmed: boolean) {
    this.authMode = false;
    if(!confirmed) {
      return;
    }
    const ref = this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: `TARIFA CREADA CON ÉXITO`
      }
    })
    ref.onClose.subscribe(() => {
      this.salir();
    })
  }
}
