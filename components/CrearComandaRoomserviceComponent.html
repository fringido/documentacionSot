import { Component, OnInit } from '@angular/core';
import {
  FormArray,
  FormBuilder,
  FormControl,
  FormGroup,
} from '@angular/forms';
import { debounceTime, distinctUntilChanged, Subject, takeUntil } from 'rxjs';
import { ActivatedRoute, Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';

import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { controlFormMixto } from 'src/app/core/formsControl/controlFormMixto';
import { ObtenerHabitacionesService } from 'src/app/core/services/habitaciones/obtener-habitaciones.service';
import { GetArticulosTodoComandaQuery ,CreateComandaInput, HabitacionDetalladaQuery} from 'src/app/core/graphQL/serviciosGraph';
import { ProductosRSService } from 'src/app/cocinabar/services/comanda/productos-rs.service';
import { SendComandaService } from 'src/app/cocinabar/services/comanda/SendComanda.service';
/**
 * Modelo creado para los articulos
 */
interface Articulos {
  id: string;
  nombre: string;
  priceWhitOutIva: number;
  precio: number;
}

@Component({
  selector: 'app-crear-comanda-roomservice',
  templateUrl: './crear-comanda-roomservice.component.html',
  styleUrls: ['./crear-comanda-roomservice.component.scss'],
})
export class CrearComandaRoomserviceComponent implements OnInit {
  //* VARIABLES
  //? VARIABLES PARA FUNCIONAMIENTO LÓGICO
  display = true;
  cortesia: boolean = false;
  //? -----------------------------------
  //? VARIABLES PARA LAS LISTAS DE PRODUCTOS Y FILTRO
  filterText!: FormControl;
  food: GetArticulosTodoComandaQuery["alimentos"] = [];
  drinks: GetArticulosTodoComandaQuery["tragos"] = [];
  sexyspa :{ __typename?: "ArticuloRs" | undefined; nombre: string; id: string;almacenId: string; precio: number; }[]= []
  other:{ __typename?: "ArticuloRs" | undefined; nombre: string; id: string;almacenId: string; precio: number; }[]= []
  foodAll: GetArticulosTodoComandaQuery["alimentos"] = [];
  drinksAll : GetArticulosTodoComandaQuery["tragos"] = [];
  otherAll:{ __typename?: "ArticuloRs" | undefined; nombre: string; id: string;almacenId: string; precio: number; }[]= []
  sexyspaAll:{ __typename?: "ArticuloRs" | undefined; nombre: string; id: string;almacenId: string; precio: number; }[]= []
/**
 * Inisialisacion del formulario tipandolo
 */
  form!: FormGroup<{
    comanda: FormArray<FormControl<unknown>>;
    cortesia: FormArray<FormControl<unknown>>;
    descuento: FormControl<number | null>;
    subTotal: FormControl<number | null>;
    total: FormControl<number | null>;
}>;
/**
 *  Inisialisacion de la estructura que se va a enviar
 */
  comanda: CreateComandaInput = {
    alimentos: [],
    bebidas: {
      articulos: [],
      tragos: []
    },
    otros: [],
    sexSpa: [],
    rentaId: ''
  }
  //? -----------------------------------
  //? VARIABLES DE FORMULARIO
  total: number = 0;
  subTotal: number = 0;
  descuento: number = 0;
  //? -----------------------------------
  //* -----------------------------------
  productos = {
    otros: [

    ],
    sexSpa: [

    ],
  };


  numHabitacion: string;

  habitacion!: HabitacionDetalladaQuery["habitacion"];
  habitacion$ = this.habitacionService.habitacionDetallada$;

  articulos!: GetArticulosTodoComandaQuery

  destroy$: Subject<boolean> = new Subject<boolean>();

/**
 * Constructor
 */
  constructor(
    private router: Router,
    private route: ActivatedRoute,
    private fb: FormBuilder,
    private dialogService: DialogService,
    private habitacionService: ObtenerHabitacionesService,
    private getArticulosService: ProductosRSService,
    private crearComanda: SendComandaService
  ) {
    this.numHabitacion = String(this.route.snapshot.paramMap.get('roomNumber'));
    this.getInfo(this.numHabitacion);
    this.form = this.fb.group({
      comanda: this.fb.array([]),
      cortesia: this.fb.array([]),
      descuento: [this.descuento],
      subTotal: [this.total],
      total: [this.subTotal],
    });

  }
  /**
   * Optiene la informacion de un servicio
 * @param numHabitacion: string
 * Numero de la habitacion sacado de router
*/
  getInfo(numHabitacion: string) {
    this.habitacionService.getHabitacio(numHabitacion);
    this.habitacionService.getHabitacioDetalle(numHabitacion)
    this.getArticulosService.getArticculos();
  }

  //* trae los datos de el FormArray
  /**
   * trae los datos de el FormArray de comanda
   */
  get comandaField() {
    return this.form.get('comanda') as FormArray;
  }
  /**
   * trae los datos de el FormArray de cortesia
   */
  get cortesiaField() {
    return this.form.get('cortesia') as FormArray;
  }
  //* ------------------------------------------
  ngOnInit(): void {

    this.initList();
    this.sumatoriaTotal();
    this.search();
  }


  //* INICIANDO LAS LISTAS
  /**
   * INICIANDO LAS LISTAS
   */
  initList() {

    this.getArticulosService.articulosComanda$.pipe(takeUntil(this.destroy$)).subscribe(articulos => {
      this.articulos = articulos;

    this.food = this.articulos.alimentos;
    this.drinks = this.articulos.tragos;

    articulos.almacenes.forEach(almacen => {
      console.log(almacen)
      if(almacen.menu == "Otros"){
        this.other = almacen.articulosAlmacen.map(d => {
          return {
          ...d.articuloRs,
          almacenId: almacen.id
        }})
      }
      if(almacen.menu == "SexSpa"){
        this.sexyspa = almacen.articulosAlmacen.map(d => {
          return {
          ...d.articuloRs,
          almacenId: almacen.id
        }})
      }

    })
    console.log(this.sexyspa, this.other)

    this.foodAll = this.food;
    this.drinksAll = this.drinks;
    this.otherAll = this.other;
    this.sexyspaAll = this.sexyspa;

    this.habitacion$.pipe(takeUntil(this.destroy$)).subscribe(data => {
      this.habitacion = data
    })
    });


  }
  //* ------------------------------------------
  //* FUNCION DE BUSQUEDA DE PRODUCTO
  /**
   * FUNCION DE BUSQUEDA DE PRODUCTO
   */
  search() {
    this.filterText = new FormControl('');
    this.filterText.valueChanges
      .pipe(debounceTime(400), distinctUntilChanged())
      .subscribe((v) => {
        if (v) {
          const filter = new RegExp(v, 'i');
          const filterFields = ['nombre'];
          this.food = this.foodAll.filter((role: any) =>
            filterFields.some((field) => filter.test(role[field]))
          );
          this.drinks = this.drinksAll.filter((role: any) =>
            filterFields.some((field) => filter.test(role[field]))
          );
          this.other = this.otherAll.filter((role: any) =>
            filterFields.some((field) => filter.test(role[field]))
          );
          this.sexyspa = this.sexyspaAll.filter((role: any) =>
            filterFields.some((field) => filter.test(role[field]))
          );
        } else {
          this.food = this.foodAll;
          this.drinks = this.drinksAll;
          this.other = this.otherAll;
          this.sexyspa = this.sexyspaAll;
        }
      });
  }
  //* ------------------------------------------
  //* SUMA PARA SACAR EL SUBTOTAL Y RESTA PARA SACAR EL TOTAL
  /**
   * SUMA PARA SACAR EL SUBTOTAL Y RESTA PARA SACAR EL TOTAL
   */
  sumatoriaTotal() {
    this.form.valueChanges.subscribe(() => {
      this.subTotal = 0;
      this.descuento = 0;
      this.total = 0;
      this.comandaField.value.map((d: any) => {
        this.subTotal = this.subTotal + d.precio * d.numProducts;
      });
      this.cortesiaField.value.map((d: any) => {
        this.subTotal = this.subTotal + d.precio * d.numProducts;
        this.descuento = this.descuento + d.precio * d.numProducts;
      });

      this.total = this.subTotal - this.descuento;
      console.log(this.form);
    });
  }
  //* ------------------------------------------

  //*EMPUJA UN FURMULARIO DEPENDIENDO DE EL ESTADO DEL BOTON CORTESIA
  /**
   * EMPUJA UN FURMULARIO DEPENDIENDO DE EL ESTADO DEL BOTON CORTESIA
   * @param product producto seleccionado
   * @param tipo que tipo de producto es (Bar, cocina, otros, sexYspa)
   */
  selectProduct(product: any, tipo: string) {
    let inList = false;
    this.comandaField.value.forEach((d: any, i: number) => {
      if (d.id == product.id) {
        this.comandaField.at(i).patchValue({ numProducts: d.numProducts + 1 });
        inList = true;
        return;
      }
      inList = false;
    });

    this.cortesiaField.value.forEach((d: any, i: number) => {
      if (d.id == product.id) {
        this.cortesiaField.at(i).patchValue({ numProducts: d.numProducts + 1 });
        inList = true;
        return;
      }
      inList = false;
    });

    if (!inList) {
      if (this.cortesia || product.precio == 0) {
        this.cortesiaField.push(
          this.fb.group({
            id: [product.id],
            tipo: [tipo],
            almacenId:[product.almacenId],
            colapse: [true],
            numProducts: [1, [controlFormMixto.isNotCero]],
            observaciones: [''],
            nombre: [product.nombre],
            precio: [product.precio],
            cortesia: [true]
          })
        );
      } else {
        this.comandaField.push(
          this.fb.group({
            id: [product.id],
            tipo: [tipo],
            almacenId:[product.almacenId],
            colapse: [true],
            numProducts: [1, [controlFormMixto.isNotCero]],
            observaciones: [''],
            nombre: [product.nombre],
            precio: [product.precio],
            cortesia: [false]
          })
        );
      }
    }
  }
  //* ------------------------------------------
  //*BOTONES DE MODAL
  /**
   * BOTONES DE MODAL SALIR
   */
  salir() {
    this.router.navigate([`/hotel`]);
  }
  /**
   * BOTONES DE MODAL ACEPTAR
   * CREA EL MODELO A ENVIA A LA MUTACION DE CREAR COMANDA
   */
  aceptar() {

    const form = this.form.getRawValue()
    console.log(form);

    const alimnetos: CreateComandaInput["alimentos"] = []
    const tragos: CreateComandaInput["bebidas"] = {
      articulos: [],
      tragos: []
    }
    const otros: CreateComandaInput["otros"] = []
    const sexySpa: CreateComandaInput["sexSpa"] = []

    if( form.comanda.length > 0 ){
      form.comanda.forEach( (el:any) => {
        console.log(el)
        if(el.tipo == 'food'){
          this.comanda.alimentos.push({
            alimentoId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: false
          })
        }

        if(el.tipo == 'drinks'){
          this.comanda.bebidas.tragos.push({
            tragoId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: false
          })
          tragos.articulos = [];
        }
        if(el.tipo == 'other'){
          this.comanda.otros.push({
            almacenId: String(el.almacenId),
            articuloId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: false
          })
        }
        if(el.tipo == 'sexyspa'){
          this.comanda.sexSpa.push({
            almacenId: String(el.almacenId),
            articuloId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: false
          })
        }
      })
    }
    if( form.cortesia.length > 0 ){

      form.cortesia.forEach( (el:any) => {
        if(el.tipo == 'food'){
          this.comanda.alimentos.push({
            alimentoId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: true
          })
        }

        if(el.tipo == 'drinks'){
          this.comanda.bebidas.tragos.push({
            tragoId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: true
          })
          tragos.articulos = [];
        }
        if(el.tipo == 'other'){
          this.comanda.otros.push({
            almacenId: String(el.almacenId),
            articuloId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: true
          })
        }
        if(el.tipo == 'sexyspa'){
          this.comanda.sexSpa.push({
            almacenId: String(el.almacenId),
            articuloId: String(el.id),
            cantidad: Number(el.numProducts),
            comentario: String(el.observaciones),
            cortesia: true
          })
        }
      })
    }

    this.comanda.rentaId = this.habitacion.rentas[0].id


    console.log(this.comanda)
    this.crearComanda.crearComnada(this.comanda)

    const ref = this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: `COMANDA GENERADA CON EXITO`,
      },
    });

    this.router.navigate([`/hotel`]);
  }


  //* ------------------------------------------
}
