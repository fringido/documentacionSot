import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { ControlDias } from 'src/app/core/formsControl/controlGastos';

@Component({
  selector: 'app-modificar-gastos-generales',
  templateUrl: './modificar-gastos-generales.component.html',
  styleUrls: ['./modificar-gastos-generales.component.scss'],
})
export class ModificarGastosGeneralesComponent implements OnInit {
  display = true;
  form!: FormGroup;
  authMode = false;

  familiaGastos = [
    {
      nombre: 'GASTOS DE OPERACIÓNl',
      id: 1,
    },
    {
      nombre: 'GASTOS DE ADMINISTRACIÓN',
      id: 2,
    },
    {
      nombre: 'GASTOS DE NOMINA',
      id: 3,
    },
  ];

  constructor(
    private fb: FormBuilder,
    private dialogService: DialogService,
    private router: Router
  ) {}

  ngOnInit(): void {
    this.creteForm();
  }
  creteForm() {
    this.form = this.fb.group({
      concepto: ['CONCEPTO 1', [Validators.required]],
      familia: [0, [Validators.required]],
      recurrencia: ['', [Validators.required, ControlDias.daysValidator]],
      caja: [false, [Validators.required]],
    });

    this.form.patchValue({
      recurrencia: 'mensual',
      caja: true,
    });
  }

  cambio() {
    if (this.form.value.caja == false) {
      this.form.patchValue({
        caja: true,
      });
    } else {
      this.form.patchValue({
        caja: false,
      });
    }
  }

  cancelar(selsct: boolean) {
    if (selsct) {
      this.router.navigate([`/hotel/gastos/gastosGenerales`]);
      this.dialogService.open(MessageModalAutoclosableComponent, {
        data: {
          message: `Gasto “${this.form.value.concepto}” Modificado con éxito`,
        },
      });
    } else {
      this.authMode = false;
    }
  }
}
