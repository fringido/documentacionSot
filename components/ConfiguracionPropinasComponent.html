import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { controlFormMixto } from 'src/app/core/formsControl/controlFormMixto';

@Component({
  selector: 'app-configuracion-propinas',
  templateUrl: './configuracion-propinas.component.html',
  styleUrls: ['./configuracion-propinas.component.scss']
})
export class ConfiguracionPropinasComponent implements OnInit {

  display = true
  form!: FormGroup;
  constructor(
    private fb :FormBuilder,
    private dialogService: DialogService,
    private readonly router: Router,
  ) {
  }

  ngOnInit(
    ) {
    this.createForm()
  }

  aceptar(){
    const genDialogMessage = (message: string) => {
      return this.dialogService.open(MessageModalAutoclosableComponent, {data: { message }});
    };
    genDialogMessage(`CONFIGURACIÓN GUARDADA CON ÉXITO`)
    this.router.navigate([`/hotel`]);

  }
  createForm(){
    this.form = this.fb.group({
      propinas: this.fb.group({
        alimentos: [null,[controlFormMixto.isNotCero,Validators.required]],
        bebidas: [null,[controlFormMixto.isNotCero]],
        sexYspa:[null,[controlFormMixto.isNotCero]],
        otros:[null,[controlFormMixto.isNotCero]],
      }),
      descuentos: this.fb.group({
        visa:[null,[controlFormMixto.isNotCero]],
        masterCard:[null,[controlFormMixto.isNotCero]],
        amex:[null,[controlFormMixto.isNotCero]],
      }),
      vendedoresYValets:this.fb.group({
        lunes:[0,[controlFormMixto.isNotCero]],
        martes:[0,[controlFormMixto.isNotCero]],
        miercoles:[0,[controlFormMixto.isNotCero]],
        jueves:[0,[controlFormMixto.isNotCero]],
        viernes:[0,[controlFormMixto.isNotCero]],
        sabado:[0,[controlFormMixto.isNotCero]],
        domingo:[0,[controlFormMixto.isNotCero]],
      })
    })

    }
}
