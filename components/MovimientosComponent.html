import { Component, OnInit, ViewChildren, QueryList, ElementRef, Renderer2 } from '@angular/core';
import { Location } from '@angular/common';
import { FormControl } from '@angular/forms';
import { debounceTime, distinctUntilChanged } from 'rxjs';
import { listaMovimientos } from './constants/movimientos-list';
import { Router } from '@angular/router';

@Component({
  selector: 'app-movimientos',
  templateUrl: './movimientos.component.html',
  styleUrls: ['./movimientos.component.scss']
})
export class MovimientosComponent implements OnInit {

  visible = true;

  movimientos!: any[];
  movimientosAll!: any[];
  @ViewChildren('movimientosRef') movimientosRef!: QueryList<ElementRef<HTMLDivElement>>;

  filterText!: any;
  isMovimientoSeleccionado: boolean = false;
  movimientoSeleccionado!: any;
  constructor(
    private readonly location: Location,
    private readonly renderer: Renderer2,
    private readonly router: Router
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
    this.search();
  }

  liberarVariables() {
    this.movimientosAll = [...listaMovimientos];
    this.movimientos = [...listaMovimientos];
  }

  search() {
    this.filterText = new FormControl("");
    this.filterText.valueChanges
      .pipe(debounceTime(400), distinctUntilChanged())
      .subscribe((v: string) => {
        this.unselectMovimientos();
        if (v) {
          this.movimientos = this.movimientosAll.filter((movimiento:any) => {
            return (movimiento.nombre).toLowerCase().includes(this.filterText.value.toLowerCase())
          }
          );
        } else {
          this.movimientos = this.movimientosAll;
        }
      });
  }

  selectMovimiento(movimiento: any, index: number) {
    if((this.movimientoSeleccionado?.id === movimiento.id) && this.isMovimientoSeleccionado) {
      return this.unselectMovimientos();
    }
    this.unselectMovimientos();
    this.movimientoSeleccionado = movimiento;
    
    const movimientoSeleccionado = this.movimientosRef?.toArray().find((el) => el.nativeElement.id === 'movimiento_' + index)?.nativeElement;
    console.log(movimientoSeleccionado);

    this.renderer.addClass(movimientoSeleccionado, 'box_componente_selected');
    this.renderer.removeClass(movimientoSeleccionado, 'box_componente');
    this.isMovimientoSeleccionado = true;
  }

  unselectMovimientos() {
    this.isMovimientoSeleccionado = false;
    this.movimientosRef?.toArray().forEach(el => {
      this.renderer.removeClass(el.nativeElement, 'box_componente_selected');
      this.renderer.addClass(el.nativeElement, 'box_componente');
    })
  }

  salir() {
    this.location.back();
  }

  redirect(path: string) {
    this.router.navigateByUrl(path);
  }
}
