import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { Subject, Subscription, takeUntil } from 'rxjs';

import { GetComandaService } from 'src/app/cocinabar/services/comanda/getComanda.service';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { GetAllComandasInfoQuery } from 'src/app/core/graphQL/serviciosGraph';



@Component({
  selector: 'app-liberar-pagar-room-service',
  templateUrl: './liberar-pagar-room-service.component.html',
  styleUrls: ['./liberar-pagar-room-service.component.scss']
})
export class LiberarPagarRoomServiceComponent implements OnInit {
  //* VARIABLES
  display = true
  //* ---------------
  comandas: { id: string; tipo: string; numeroOrden: number; fechaSolicitud: any; estadoPedido: string; articulos: { cantidad: number; nombre: string; costo: number; esCortesia: boolean; }[]; subtotal: number; descuemto: number; cortesia: number; total: number; }[] = []

  destroy$: Subject<boolean> = new Subject<boolean>();

  // Variables provisionales la habitacion seleccionada se va a obtener por medio de los params en la ruta

  authMode = false
  idComandaSelect: any;

  comandasData: GetAllComandasInfoQuery["comandas"] = []

  comandas$ = this.getComandaService.comandas$

  numHabitacion: string
  constructor(
    private router: Router,
    private route: ActivatedRoute,
    public dialogService: DialogService,
    private getComandaService: GetComandaService,

  ) {
    this.numHabitacion = String(this.route.snapshot.paramMap.get("roomNumber"));
    this.getComandaService.getComandas()
    this.getComandaService.getComandasSuscri();
  }

  ngOnInit(): void {
    this.comandas$.pipe(takeUntil(this.destroy$)).subscribe(data => {
      this.comandasData = data.filter(data => data.renta.habitacion.numeroHabitacion == Number(this.numHabitacion))
      this.formatoComanda();
    })


  }


  formatoComanda() {
    this.comandas = []

    this.comandasData.forEach((data, i) => {

      const comanda: { cantidad: number; nombre: string; costo: number; esCortesia: boolean; }[] = []
      let subtotal: number = 0
      let cortesia: number = 0
      let estado = 'Preparación'
      let tipo = ''

      if (data.fechaPago == null) {
        console.log(data,'comandas')
        if (data.alimentosComanda.length > 0 && data.tragosComanda.length > 0) {
          tipo = 'AB'
          data.alimentosComanda.forEach(alimento => {
            subtotal = subtotal + alimento.alimento.precio
            comanda.push({
              cantidad: alimento.cantidad,
              nombre: alimento.alimento.nombre,
              costo: alimento.alimento.precio,
              esCortesia: alimento.cortesia,
            })
            if (alimento.cortesia) {
              cortesia = cortesia + alimento.alimento.precio
            }
          })
          data.tragosComanda.forEach(trago => {
            subtotal = subtotal + trago.trago.precio
            comanda.push({
              cantidad: trago.cantidad,
              nombre: trago.trago.nombre,
              costo: trago.trago.precio,
              esCortesia: trago.cortesia
            })
            if (trago.cortesia) {
              cortesia = cortesia + trago.trago.precio
            }
          })
          console.log(data.alimentosComanda[0].fechaPorEntregar != null && data.tragosComanda[0].fechaPorEntregar != null, 'valor')
          if (data.alimentosComanda[0].fechaPorEntregar != null && data.tragosComanda[0].fechaPorEntregar != null) {
            estado = 'Por cobrar'
          }

        }

        if (data.alimentosComanda.length > 0 && data.tragosComanda.length == 0) {
          tipo = 'A'
          data.alimentosComanda.forEach(alimento => {
            subtotal = subtotal + alimento.alimento.precio
            comanda.push({
              cantidad: alimento.cantidad,
              nombre: alimento.alimento.nombre,
              costo: alimento.alimento.precio,
              esCortesia: alimento.cortesia
            })
            if (alimento.cortesia) {
              cortesia = cortesia + alimento.alimento.precio
            }
          })
          if (data.alimentosComanda[0].fechaPorEntregar != null) {
            estado = 'Por cobrar'
          }
        }

        if (data.tragosComanda.length > 0 && data.alimentosComanda.length == 0) {
          tipo = 'B'
          data.tragosComanda.forEach(trago => {
            subtotal = subtotal + trago.trago.precio
            comanda.push({
              cantidad: trago.cantidad,
              nombre: trago.trago.nombre,
              costo: trago.trago.precio,
              esCortesia: trago.cortesia
            })
            if (trago.cortesia) {
              cortesia = cortesia + trago.trago.precio
            }
          })
          if (data.tragosComanda[0].fechaPorEntregar != null) {
            estado = 'Por cobrar'
          }
        }

        if(data.articulosComanda.length > 0) {
          console.log(data,'filtro')
          let otros = false
          let sexYSpa = false
          data.articulosComanda.forEach(articulo => {
            if(articulo.almacen.menu == 'Otros'){
              otros = true
            }
            if(articulo.almacen.menu == 'SexSpa'){
              sexYSpa = true
            }
            subtotal = subtotal + articulo.articuloRs.precio
            comanda.push({
              cantidad: articulo.cantidad,
              nombre: articulo.articuloRs.nombre,
              costo: articulo.articuloRs.precio,
              esCortesia: articulo.cortesia
            })
            if (articulo.cortesia) {
              cortesia = cortesia + articulo.articuloRs.precio
            }
          })
          if(otros){
            tipo = `${tipo}O`
          }
          if(sexYSpa){
            tipo = `${tipo}S`
          }
          if (data.tragosComanda.length == 0 && data.alimentosComanda.length == 0){
            estado = 'Por cobrar'
          }
        }

        if (data.fechaEntrega) {
          estado = 'Por pagar'
          this.comandas.push({
            id: data.id,
            tipo: tipo,
            numeroOrden: i,
            fechaSolicitud: data.fechaEntrega,
            estadoPedido: estado,
            articulos: comanda,
            subtotal: subtotal,
            descuemto: 0,
            cortesia: cortesia,
            total: subtotal - cortesia
          })
        } else {
          this.comandas.push({
            id: data.id,
            tipo: tipo,
            numeroOrden: i,
            fechaSolicitud: data.fechaSolicitud,
            estadoPedido: estado,
            articulos: comanda,
            subtotal: subtotal,
            descuemto: 0,
            cortesia: cortesia,
            total: subtotal - cortesia
          })
        }
      }
    })
    this.comandas.sort((a, b) => Number(b.numeroOrden) -  Number(a.numeroOrden));
  }



  //* BOTONES MODAL
  salir() {
    this.router.navigate([`/hotel`]);
  }


  ngOnDestroy() {
    this.comandas$.pipe(takeUntil(this.destroy$)).subscribe().unsubscribe()
  }
  //* ------------------------------------------
}
