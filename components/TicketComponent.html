import { Component, OnInit, Renderer2, ElementRef, ViewChild, Input, Output, EventEmitter } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { validaciones } from 'src/app/core/formsControl/validaciones';
import { PrimengModule } from 'src/app/core/modules/primeng/primeng.module';
import { TicketsImpresora } from '../../../constants/tickets-list';
import { TicketsService } from '../../../services/tickets/tickets.service';

@Component({
  selector: 'app-ticket',
  templateUrl: './ticket.component.html',
  styleUrls: ['./ticket.component.scss'],
  standalone: true,
  imports: [PrimengModule, FormsModule, CommonModule]
})
export class TicketComponent implements OnInit {

  @ViewChild('ticketTemplate') ticketTemplate!: ElementRef<HTMLDivElement>;
  @ViewChild('botonBorrar') botonBorrar!: ElementRef<HTMLButtonElement>;



  @Output('ticketChanged') ticketChanged = new EventEmitter<boolean>();
  @Input('ticket') ticket!: TicketsImpresora;
  @Input('isEditable') isEditable: boolean = true;

  readonly validaciones = new validaciones();


  constructor(
    private readonly renderer: Renderer2,
    private readonly ticketsService: TicketsService
  ) { }

  ngOnInit(): void {

  }

  toggle(event: MouseEvent) {
    event.stopPropagation();
    this.ticketTemplate?.nativeElement.classList.toggle('d-none');
    this.botonBorrar?.nativeElement.classList.toggle('d-none');
    
  }

  deleteConfig(ticket: TicketsImpresora) {
    console.log(ticket);
    
    this.ticketsService.removeTicket(ticket);
  }

  onChange() {
    this.ticketChanged.emit(true)
    console.log('changed');
    
  }

}
