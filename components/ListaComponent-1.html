import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { DialogService } from 'primeng/dynamicdialog';
import { debounceTime, distinctUntilChanged } from 'rxjs/operators';
import { ConfimModalMessageComponent } from 'src/app/core/components/confim-modal-message/confim-modal-message.component';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { validaciones } from 'src/app/core/formsControl/validaciones';
import { MatriculaService } from '../../services/matricula/matricula.service';
import { listaReportes } from './constants/lista-reportes';

@Component({
  selector: 'app-lista',
  templateUrl: './lista.component.html',
  styleUrls: ['./lista.component.scss']
})
export class ListaComponent implements OnInit {

  filterText!: FormControl;
  validacionesFormulario!: validaciones;

  showReporte!: boolean;
  listaReportes!: any[];
  idSelect!: string;
  matricula!: string;
  inavilitarReporte!: boolean;

  constructor(
    private readonly matriculaService: MatriculaService,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.search()
    this.liberarVariables();
  }

  liberarVariables() {
    this.showReporte = true;
    this.listaReportes = [];
    this.idSelect = '';
    this.matricula = '';
    this.inavilitarReporte = false;
    this.validacionesFormulario = new validaciones();
  }

  search() {
    this.filterText = new FormControl("");
    this.filterText.valueChanges
      .pipe(debounceTime(400), distinctUntilChanged())
      .subscribe((v) => {
        this.matriculaService.updateMatricula(v);
        if(this.filterText.value.length < 6) {
          this.listaReportes = [];
          this.showReporte = true;
          return;
        }
        // TODO: http a getReportes
        // subscribe(reportes) => {
          this.listaReportes = [...listaReportes]
          // TODO: mostrar en rojo los reportes inhabilitados
          this.showReporte = true;
          if(!this.listaReportes.length) {
            this.showReporte = false;
          }
        // }
      });
  }

  selectMatricula(reporte: any) {
    this.idSelect = reporte.id
    this.matriculaService.updateMatriculaDetails(reporte);
    this.matriculaService.updateMatricula(reporte.matricula);
    this.inavilitarReporte = reporte.activo ? true : false;
  }

  inHabilitarReporte() {
    const ref = this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: '¿DESEAS INHABILITAR EL REPORTE?'
      }
    });

    ref.onClose.subscribe(({confirmed}) => {
      if(!confirmed) {
        return;
      }
      this.matriculaService.updateAuthMode(true);
    });
  }
  
}
