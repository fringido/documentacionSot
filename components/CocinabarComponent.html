import { Component, OnDestroy, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { BehaviorSubject, Observable, Subject, of } from 'rxjs';
import { take, tap, takeUntil, mergeWith, map, catchError } from 'rxjs/operators';
import { MessageModalAutoclosableComponent } from '../core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { GetComandasBarYCocinaQuery } from '../core/graphQL/serviciosGraph';
import { CocinabarService } from './services/cocinabar/cocinabar.service';
import { GetComandaService } from './services/comanda/getComanda.service';
import { LiberarCoandaService } from './services/comanda/liberarCoanda.service';
import { ComandaPagoEnvioService } from './services/comanda/comandaPagoEnvio.service';
@Component({
  selector: 'app-cocinabar',
  templateUrl: './cocinabar.component.html',
  styleUrls: ['./cocinabar.component.scss']
})
export class CocinabarComponent implements OnInit, OnDestroy {

  private destroy$ = new Subject<boolean>();
  loader = true
  authMode = {
    porPreparar: false,
    porEntregar: false,
  }

  isValidNumber(e: KeyboardEvent) {
    e = e || window.event;
    let charCode = (typeof e.which == "undefined") ? e.keyCode : e.which;
    let charStr = String.fromCharCode(charCode);
    if (!charStr.match(/^[0-9]+$/)) {
      e.preventDefault();
    }
  }

  constructor(
    private readonly cocinabarService: CocinabarService,
    private readonly activatedRoute: ActivatedRoute,
    private readonly router: Router,
    private comandasService: GetComandaService,
    private readonly fb: FormBuilder,
    private dialogService: DialogService,
    private liberarComandaService: LiberarCoandaService,
    private readonly pagoComandaService: ComandaPagoEnvioService
  ) {
    this.form = this.fb.group({
      name: ['', [Validators.required]],
      password: ['', [Validators.required]]
    });
  }
  ngOnDestroy(): void {
    this.destroy$.next(true);
  }

  ngOnInit(): void {
    this.tipoComanda = (this.activatedRoute.data as BehaviorSubject<any>).value.path;
    this.tipoComandaItems = this.tipoComanda === 'bar' ? 'tragosComanda' : 'alimentosComanda';
    this.tipoComanda === 'bar' ? 
        this.comandasService.getComandasBarSuscri().subscribe():
        this.comandasService.getComandasCocinaSuscri().subscribe();
    this.cocinabarService.getCocinaBarData(this.tipoComanda).pipe(
      take(1)
    ).subscribe(({comandas}) => {
      this.setComandas(comandas);
    })
    this.cocinabarService.getCocinaBarSubs(this.tipoComanda).pipe(
      takeUntil(this.destroy$)
    ).subscribe(comandas => {
      this.setComandas(comandas);
    });
  }

  setComandas(comandas: any[]) {
    this.comandas = comandas;
    // TODO: la peticion trae todas las comandas, sin importar que ya sean entregadas y pagadas, deberia traer solo las que estan por preparar y por entregar
    console.log(comandas);
    
    const commands = comandas.filter((comanda: any) => comanda[this.tipoComandaItems].length && !comanda.fechaPago)
    this.comandasPreparar = commands.filter((comanda: any) => !comanda[this.tipoComandaItems][0].fechaPorEntregar);
    this.comandasEntregar = commands.filter((comanda: any) => ((comanda[this.tipoComandaItems][0].fechaPorEntregar) && (!comanda.fechaEntrega)));
  }

  selectPorPreparar(id: string) {
    this.idComandaSelect = id;
    this.authMode.porPreparar = true
  }

  selectPorEntregar(id: string) {
    this.idComandaSelect = id;
    this.authMode.porEntregar = true
  }

  goBack() {
    localStorage.getItem('token') ?
      this.router.navigate(['hotel']) :
      this.router.navigate(['login'])
  }

  liberarComandaPorPreparar(confirmed: boolean){
    this.authMode.porPreparar = false;
    if(!confirmed) {
      return;
    }
    
    let mensaje
    if(this.tipoComanda == 'bar'){
      this.liberarComandaService.liberarBarra(this.idComandaSelect)
    }
    if(this.tipoComanda == 'cocina'){
      this.liberarComandaService.liberarCocina(this.idComandaSelect)
    }
    this.liberarComandaService.mensaje$.pipe(take(1)).subscribe({
      next: data => {
        return this.dialogService.open(MessageModalAutoclosableComponent, {
          data: {
            message: 'COMANDA PREPARADA CON ÉXITO'
          },
        });
      },
      error: err => {
        return this.dialogService.open(MessageModalAutoclosableComponent, {
          data: {
            message: 'OCURRIÓ UN ERROR AL PREPARAR LA COMANDA'
          },
        });
      }
    })
  }

  liberarComandaPorEntregar(confirmed: boolean) {
    if(!confirmed) {
      return;
    }
    this.authMode.porEntregar = false;
    this.pagoComandaService.entregarComanda(this.idComandaSelect).subscribe({
      next: res => {
      this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: 'COMANDA ENVIADA CON ÉXITO'
        }
      })
    }, 
    error: err => {
      this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: 'OCURRIÓ UN ERROR AL ENVIAR LA COMANDA'
        }
      })
    }
    });
  }
}
