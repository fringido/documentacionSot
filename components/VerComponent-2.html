import { Location } from '@angular/common';
import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { areas, impresoras } from '../../constants/impresora-detalles';

@Component({
  selector: 'app-ver',
  templateUrl: './ver.component.html',
  styleUrls: ['./ver.component.scss']
})
export class VerComponent implements OnInit {

  visible = true;

  impresora: FormGroup = this.fb.group({
    tipo: ['', Validators.required],
    area: ['', Validators.required],
    name: ['', Validators.required]
  });

  get tipo() { return this.impresora.get('tipo'); }
  get area() { return this.impresora.get('area'); }
  get name() { return this.impresora.get('name'); }

  areas!: any[];
  impresorasPreregistradas!: any[];

  constructor(
    private readonly location: Location,
    private readonly fb: FormBuilder
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
    this.impresora.disable();
  }

  liberarVariables() {
    this.areas = [...areas];
    this.impresorasPreregistradas = [...impresoras];
    this.impresora.setValue({
      tipo: this.impresorasPreregistradas[0].nombre_en_red,
      area: this.areas[0].nombre,
      name: 'Ejemplo'
    });
  }

  salir() {
    this.location.back();
  }
}
