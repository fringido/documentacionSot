import { Component, ElementRef, OnInit, QueryList, Renderer2, ViewChildren } from '@angular/core';
import { Location } from '@angular/common'
import { Router } from '@angular/router';
import { impresorasList } from 'src/app/impresoras/lista/constants/impresoras';
import { FormControl } from '@angular/forms';
import { debounceTime, distinctUntilChanged } from 'rxjs/operators';
import { TicketsService } from 'src/app/impresoras/movimientos/services/tickets/tickets.service';
import { TicketsImpresora } from 'src/app/impresoras/movimientos/constants/tickets-list';

@Component({
  selector: 'app-select-impresora',
  templateUrl: './select-impresora.component.html',
  styleUrls: ['./select-impresora.component.scss']
})
export class SelectImpresoraComponent implements OnInit {

  visible = true;

  impresoras!: any[];
  impresorasAll!: any[];
  @ViewChildren('impresorasRef') impresorasRef!: QueryList<ElementRef<HTMLDivElement>>;

  redirectTimeout!: NodeJS.Timeout;

  filterText!: any;
  isImpresoraSeleccionada: boolean = false;
  impresoraSeleccionada!: any;
  constructor(
    private readonly location: Location,
    private readonly renderer: Renderer2,
    private readonly router: Router,
    private readonly ticketsService: TicketsService
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
    this.search();
  }

  liberarVariables() {
    this.impresorasAll = [...impresorasList];
    this.impresoras = [...impresorasList];
  }

  search() {
    this.filterText = new FormControl("");
    this.filterText.valueChanges
      .pipe(debounceTime(400), distinctUntilChanged())
      .subscribe((v: string) => {
        this.unselectImpresoras();
        if (v) {
          this.impresoras = this.impresorasAll.filter((impresora:any) => {
            return (impresora.nombre).toLowerCase().includes(this.filterText.value.toLowerCase())
          }
          );
        } else {
          this.impresoras = this.impresorasAll;
        }
      });
  }

  selectImpresora(impresora: any, index: number) {
    if((this.impresoraSeleccionada?.id === impresora.id) && this.impresoraSeleccionada) {
      return this.unselectImpresoras();
    }
    this.unselectImpresoras();
    this.impresoraSeleccionada = impresora;
    
    const impresoraSeleccionada = this.impresorasRef?.toArray().find((el) => el.nativeElement.id === 'impresora_' + index)?.nativeElement;
    console.log(impresoraSeleccionada);

    this.renderer.addClass(impresoraSeleccionada, 'box_componente_selected');
    this.renderer.removeClass(impresoraSeleccionada, 'box_componente');
    this.isImpresoraSeleccionada = true;

    this.redirectTimeout = setTimeout(() => {
      this.addTicket();
      this.salir();
    }, 500)
  }

  unselectImpresoras() {
    this.isImpresoraSeleccionada = false;
    this.impresoraSeleccionada = null;
    this.impresorasRef?.toArray().forEach(el => {
      this.renderer.removeClass(el.nativeElement, 'box_componente_selected');
      this.renderer.addClass(el.nativeElement, 'box_componente');
    });
    clearTimeout(this.redirectTimeout);
  }

  addTicket() {
    this.ticketsService.addTicket({
      impresora: {
        name: this.impresoraSeleccionada.name,
        id: this.impresoraSeleccionada.id
      },
      movimiento: {
        name: 'TODO',
        id: '123-123'
      },
      tickets: {
        miniticket: 0,
        ticket: 0
      }
    });
  }

  salir() {
    this.ticketsService.updateModoSeleccionarImpresora(false);
  }
}
