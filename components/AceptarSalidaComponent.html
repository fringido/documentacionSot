import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { Subject, take, takeUntil } from 'rxjs';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import {
  RentaQuery,
  ValidEstadosHabitaciones,
} from 'src/app/core/graphQL/serviciosGraph';
import { ExtrasDeHabitacionService } from 'src/app/core/services/rentaHabitaicones/extrasDeHabitacion.service';
import { GetRentaService } from 'src/app/core/services/rentaHabitaicones/getRenta.service';
import { RoomStatusEnum } from 'src/app/room-types/enums/room-status.enum';
import { Room } from 'src/app/room-types/interfaces/room.interface';
import { RoomService } from 'src/app/room-types/services/room/room.service';

@Component({
  selector: 'app-aceptar-salida',
  templateUrl: './aceptar-salida.component.html',
  styleUrls: ['./aceptar-salida.component.scss'],
})
export class AceptarSalidaComponent implements OnInit {
  display = true;

  rentas$ = this.getRentas.rentas$;
  renta!: RentaQuery['rentas'];
  selectedRoom!: Room;

  numeroHabitacion: string = '';

  destroy$: Subject<boolean> = new Subject<boolean>();


  constructor(
    private readonly roomService: RoomService,
    private readonly router: Router,
    public dialogService: DialogService,
    private route: ActivatedRoute,
    private getRentas: GetRentaService,
    private extrasDeHabitacion: ExtrasDeHabitacionService
  ) {
    this.numeroHabitacion = String(this.route.snapshot.paramMap.get('roomNumber'));
    this.getRentas.getRenta();
  }

  ngOnInit(): void {
    this.getInfo();
  }

  getInfo() {
    this.rentas$.pipe(takeUntil(this.destroy$)).subscribe((data) => {
        this.renta = data.filter(
          (renta) => renta.habitacion.numeroHabitacion == Number(this.numeroHabitacion)
        );
        this.salida();
      })
  }

  salida() {
    this.extrasDeHabitacion.salidaHabitacion(this.renta[0].id);

    const genDialogMessage = (message: string) => {
      return this.dialogService.open(MessageModalAutoclosableComponent, {
        data: { message },
      });
    };
    genDialogMessage(`RENTA FINALIZADA`);
    this.router.navigate([`/hotel`]);
  }
}
