import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { DialogService } from 'primeng/dynamicdialog';
import { ConfimModalMessageComponent } from 'src/app/core/components/confim-modal-message/confim-modal-message.component';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';

@Component({
  selector: 'app-conceptos-mantenimiento',
  templateUrl: './conceptos-mantenimiento.component.html',
  styleUrls: ['./conceptos-mantenimiento.component.scss']
})
export class ConceptosMantenimientoComponent implements OnInit {

  conceptos = [
    {
      id: 1,
      nombre: 'LIMPIEZA OK',
      tipo: 'LIBERACIÓN DE HABITACIÓN'
    },
    {
      id: 2,
      nombre: 'LIMPIEZA CON DETALLE',
      tipo: 'LIBERACIÓN DE HABITACIÓN'
    },
    {
      id: 3,
      nombre: 'LIBERACIÓN EXITOSA',
      tipo: 'LIBERACIÓN DE HABITACIÓN'
    },
    {
      id: 4,
      nombre: 'LIMPIEZA CON DETALLE DE SUPERVISOR',
      tipo: 'LIBERACIÓN DE HABITACIÓN'
    },
    {
      id: 5,
      nombre: 'HUBO DETALLE DE VENTILACIÓN',
      tipo: 'LIBERACIÓN DE HABITACIÓN'
    },
    {
      id: 6,
      nombre: 'HUBO DETALLE PORMENOR',
      tipo: 'LIBERACIÓN DE HABITACIÓN'
    },
    {
      id: 7,
      nombre: 'FUMIGACIÓN',
      tipo: 'BLOQUEO'
    },
    {
      id: 8,
      nombre: 'BLOQUEO',
      tipo: 'BLOQUEO'
    },
  ]

  concepto = {
    id: 0,
    nombre: '',
    tipo: ''
  }
  constructor(private dialogService: DialogService) {}

  display = true;
  authMode = false;
  filterText!: FormControl;

  ngOnInit(): void {
  }

  cancelar(selsct: boolean) {
    if (selsct) {
      const mensaje = 'CONCEPTO DESCARTADO CON ÉXITO';
      this.mensaje(mensaje);
      this.authMode = false;
    } else {
      this.authMode = false;
    }
  }

  mensaje(mensaje: string) {
    return this.dialogService.open(MessageModalAutoclosableComponent, {
      data: {
        message: mensaje,
      },
    });
  }

  mensajeConfirmacion(mensaje: string) {
    return this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: mensaje,
      },
    });
  }

  descartar() {
    const ref = this.mensajeConfirmacion(
      `ESTAS POR DESCARTAR EL CONCEPTO "${this.concepto.nombre}"`
    );

    ref.onClose.subscribe(({ confirmed }) => {
      if (!confirmed) {
        return;
      }
      this.authMode = true;
    });
  }
}
