import { Component, OnInit, Input, Renderer2, ViewChildren, QueryList, ElementRef, OnDestroy } from '@angular/core';
import { Tarifa, TarifasGroup } from '../../constants/tarifas-list';
import { TarifasService } from '../../services/tarifas/tarifas.service';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-card',
  templateUrl: './card.component.html',
  styleUrls: ['./card.component.scss']
})
export class CardComponent implements OnInit, OnDestroy {

  @Input('tarifasGroup') tarifasGroup!: TarifasGroup;

  @ViewChildren('tarifasRef') tarifasRef!: QueryList<ElementRef<HTMLSpanElement>>

  tarifaSeleccionada!: any; // TODO: tipar tarifa
  private readonly destroy$ = new Subject<boolean>();


  constructor(
    private readonly renderer: Renderer2,
    private readonly tarifasService: TarifasService
  ) { }

  ngOnInit(): void {
    this.tarifasService.seleccionTarifa$.pipe(
      takeUntil(this.destroy$)
    ).subscribe((value: Tarifa | null) => {
      this.deseleccionarTarifas();
    });
  }

  ngOnDestroy(): void {
    this.destroy$.next(true);
    this.destroy$.unsubscribe();
  }

  seleccionarTarifa(tarifa: Tarifa) {
    if(this.tarifaSeleccionada?.nativeElement.id === tarifa.id) {
      this.tarifasService.emitirSeleccionarTarifa(null);
      return this.deseleccionarTarifas();
    }
    this.tarifasService.emitirSeleccionarTarifa(tarifa);
    this.deseleccionarTarifas();
    this.tarifaSeleccionada = this.tarifasRef.toArray().find((tarifaEl) => tarifaEl.nativeElement.id === tarifa.id);
    this.tarifaSeleccionada?.nativeElement.classList.toggle('boxBadgeSelected');
  }

  deseleccionarTarifas() {
    this.tarifaSeleccionada = null;
    this.tarifasRef?.toArray().forEach((tarifaEl) => {
      this.renderer.removeClass(tarifaEl.nativeElement, 'boxBadgeSelected')
    });
  }
}
