import { Component, Input, OnInit } from '@angular/core';

@Component({
  selector: 'app-command',
  templateUrl: './command.component.html',
  styleUrls: ['./command.component.scss']
})
export class CommandComponent implements OnInit {

  @Input() comanda: any
  @Input() tipoComanda!: 'cocina' | 'bar';
  endDate!: Date;

  time!: string;
  countdown: any;
  timeLeft: any;

  color: any;

  items: {name: string, quantity: number, comentario: string}[] = [];

  constructor() { }

  ngOnInit(): void {

    if(this.tipoComanda === 'bar') {
      this.items = [...this.comanda.tragosComanda
        .map((trago: any) => {
        return {
          name: trago.trago.nombre || '',
          quantity: trago.cantidad || 0,
          comentario: trago.comentario
        }
      })];
    }
    if(this.tipoComanda === 'cocina') {
      this.items = [...this.comanda.alimentosComanda
        .map((alimento: any) => {
        return {
          name: alimento.alimento.nombre || '',
          quantity: alimento.cantidad || 0,
          comentario: alimento.comentario
        }
      })];
    }
    this.time = this.comanda.fechaSolicitud
  }
  cuentaRegresiba(){
    this.endDate = new Date(this.comanda.fechaSolicitud);
    this.countdown = setInterval(() => {
      const timeDiff = new Date().getTime() -  this.endDate.getTime()
      this.timeLeft = timeDiff
      if (timeDiff <= 0) {
        clearInterval(this.countdown);
        this.timeLeft = 'Expired';
      } else {
        const hours = Math.floor((timeDiff / (1000 * 60 * 60)) % 24);
        const minutes = Math.floor((timeDiff / 1000 / 60) % 60);
        const seconds = Math.floor((timeDiff / 1000) % 60);

        // this.timeLeft = `${hours}h ${minutes}m ${seconds}s`;
      }
    }, 1000);
  }
}
