import { Component, OnInit } from '@angular/core';
import { CurrencyPipe, Location } from '@angular/common';
import { FormControl, Validators } from '@angular/forms';
import { DialogService } from 'primeng/dynamicdialog';
import { ConfimModalMessageComponent } from 'src/app/core/components/confim-modal-message/confim-modal-message.component';
import { skip } from 'rxjs/operators';
import { Subscription } from 'rxjs';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';
import { validaciones } from 'src/app/core/formsControl/validaciones';


@Component({
  selector: 'app-configurar',
  templateUrl: './configurar.component.html',
  styleUrls: ['./configurar.component.scss']
})
export class ConfigurarComponent implements OnInit {

  visible = true;

  inputNumberSelected!: boolean;

  readonly validaciones = new validaciones()
  inputEventTimer!: NodeJS.Timeout;

  min: number = 1000;
  max: number = 10000;

  value: number = this.min;
  limiteValido!: boolean;

  constructor(
    private readonly location: Location,
    private readonly dialogService: DialogService
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
  }

  liberarVariables() {
    this.limiteValido = false;
    this.inputNumberSelected = false;
  }

  currencyInputChanged(value: number) {
    this.limiteValido = (value >= this.min) && (value <= this.max);
  }

  confirmarConfiguracion() {
    const ref = this.dialogService.open(ConfimModalMessageComponent, {
      data: {
        message: `ESTÁS POR MODIFICAR LA FAJILLA A $${this.value}`
      }
    });
    ref.onClose.subscribe(({confirmed}) => {
      if(!confirmed) {
        return;
      }
      this.dialogService.open(MessageModalAutoclosableComponent, {
        data: {
          message: 'SE HA MODIFICADO CON ÉXITO'
        }
      })
      this.salir();
      this.liberarVariables();
    })
  }

  salir() {
    this.location.back();
  }

}
