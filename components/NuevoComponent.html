import { Component, OnInit, OnDestroy, Output, EventEmitter } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { validaciones } from '../../../../core/formsControl/validaciones';
import { MatriculaService } from '../../services/matricula/matricula.service';

@Component({
  selector: 'app-nuevo',
  templateUrl: './nuevo.component.html',
  styleUrls: ['./nuevo.component.scss']
})
export class NuevoComponent implements OnInit, OnDestroy {

  destroy$: Subject<boolean> = new Subject<boolean>();
  @Output('nuevoReporte') nuevoReporte = new EventEmitter<{matricula: string, descripcion: string}>();

  showEnvioEntrada!: boolean;
  showNuevoReporte!: boolean;
  detalles!: string;

  descripcion!: FormControl;
  validacionesFormulario!: validaciones;



  // TODO: tipar matricula
  matricula!: string;

  constructor(
    private readonly matriculaService: MatriculaService
  ) { }

  ngOnInit(): void {
    this.liberarVariables();
    this.validarNuevoReporte();
  }

  ngOnDestroy(): void {
    this.matriculaService.updateMatricula('')
    this.destroy$.next(true);
    this.destroy$.unsubscribe();
  }

  liberarVariables() {
    this.validacionesFormulario = new validaciones();
    this.showEnvioEntrada = false
    this.showNuevoReporte = false
    this.matriculaService.matricula$
    .pipe(takeUntil(this.destroy$))
    .subscribe((value) => {
      this.matricula = value;
      if(!(value?.length > 5)) {
        this.showNuevoReporte = false
        return this.descripcion?.reset()
      }
      this.showNuevoReporte = true
    });
    this.matriculaService.matriculaDetails$
    .pipe(takeUntil(this.destroy$))
    .subscribe((matriculaDetails) => {
      this.detalles = matriculaDetails;
    });
  }

  validarNuevoReporte() {
    this.descripcion = new FormControl("");
    this.descripcion.valueChanges.subscribe((valor) => {
      this.matriculaService.updateNuevoReporteValido(false);
      if (valor?.length > 0 && this.showNuevoReporte) {
        this.detalles = valor;
        this.matriculaService.updateNuevoReporteValido(true)
        this.nuevoReporte.emit({descripcion: this.detalles, matricula: this.matricula.toUpperCase()})
      }
      if (this.matricula == "") {
        this.matriculaService.updateNuevoReporteValido(false);
        this.detalles = "";
      }
    });
  }

}
