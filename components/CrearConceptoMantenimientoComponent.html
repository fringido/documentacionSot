import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { DialogService } from 'primeng/dynamicdialog';
import { MessageModalAutoclosableComponent } from 'src/app/core/components/message-modal-autoclosable/message-modal-autoclosable.component';

@Component({
  templateUrl: './crear-concepto-mantenimiento.component.html',
  styleUrls: ['./crear-concepto-mantenimiento.component.scss']
})
export class CrearConceptoMantenimientoComponent implements OnInit {
  form!: FormGroup
  display = true
  authMode = false
  constructor(
    private fb: FormBuilder,
    private dialogService: DialogService,
    private router: Router,
  ) {
    this.createForm()
  }

  ngOnInit(): void {

  }

  /**
   * Crea el formulario
   */
  createForm() {
    this.form = this.fb.group({
      nombre: ['', [Validators.required]],
      liberarHabitacion: [true,[Validators.requiredTrue]],
      mantenimiento: [false]
    })
  }

  aceptar(event: boolean) {
    if (event) {
      const send = this.form.getRawValue()
      const genDialogMessage = (message: string) => {
        return this.dialogService.open(MessageModalAutoclosableComponent, { data: { message } });
      };
      genDialogMessage(`CONCEPTO “${send.nombre}” CREADO CON ÉXITO`)
      this.router.navigate([`/hotel/catalogos/conceptosMantenimiento`]);
    }else{
      this.authMode = false
    }
  }

}
